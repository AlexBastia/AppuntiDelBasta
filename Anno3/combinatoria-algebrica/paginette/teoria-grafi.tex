\chapter{Teoria dei grafi}

Riferimento principale: Stanley, \textit{Algebraic Combinatorics}.

\section{Introduzione}

\dfn{Grafo}{\label{def:grafo}
  Un \textit{grafo} $ G $ e' la terna $ (V, E, \varphi) $ dove $ V $ e' un insieme finito di \textit{vertici}, $ E $ e' un insieme finito di \textit{lati} e
  \[
    \varphi : E \longrightarrow \left(\!\!\binom{V}{2}\!\!\right)
  \]
  associa ad ogni lato la coppia (non ordinata) di vertici ad esso adiacenti. Qui $ \left(\!\!\binom{V}{2}\!\!\right) $ denota il \textit{multi-insieme} di taglia $ 2 $ su $ V $: ammettiamo cioe' anche coppie del tipo $ \{v, v\} $ (\textit{loop}). Tutti i grafi che consideriamo sono \textbf{non orientati}.
}

\ex{}{
  Per $ V = \{v_1, v_2, v_3\} $ con lati $ e_1, e_2, e_3 $ e $ \varphi(e_1) = \{v_1, v_2\} $: scriveremo brevemente $ \varphi(e_1) = v_1 v_2 $.
}

\dfn{Lato multiplo}{\label{def:lato-multiplo}
  Se esistono lati distinti $ e_1, \ldots, e_k \in E $ con
  \[
    \varphi(e_1) = \cdots = \varphi(e_k) = uv ,
  \]
  diciamo che fra $ u $ e $ v $ c'e' un \textit{lato multiplo}.
}

\dfn{Grafo semplice}{\label{def:grafo-semplice}
  Un grafo $ G $ si dice \textit{semplice} se non ha loops ne' lati multipli: equivalentemente, $ \varphi $ e' iniettiva e a valori in $ \binom{V}{2} $ (coppie distinte, senza ripetizione), ossia
  \[
    E \subseteq \binom{V}{2} .
  \]
}

\subsection{Matrice di adiacenza e conteggio cammini}

\subsubsection{La matrice $ A(G) $}

\dfn{Matrice di adiacenza}{\label{def:matrice-adiacenza}
  Sia $ G $ un grafo con vertici $ V = \{v_1, \ldots, v_p\} $. La \textit{matrice di adiacenza} $ A(G) \in \mathbb{N}^{p \times p} $ ha entrate
  \[
    A(G)_{ij} := \#\{\text{lati incidenti a } v_i \text{ e a } v_j\} .
  \]
}

Proprieta' immediate:
\begin{itemize}
  \item $ A(G) $ e' \textbf{simmetrica} (i grafi non sono orientati);
  \item la traccia $ \mathrm{Tr}(A(G)) = \sum_i A(G)_{ii} $ conta il numero di \textit{loops};
  \item essendo simmetrica reale, $ A(G) $ e' \textbf{diagonalizzabile} (teorema spettrale).
\end{itemize}

Vedremo come autovalori di $ A(G) $ e cammini sul grafo siano strettamente collegati.

\subsubsection{Potenze di $ A(G) $ e cammini}

\dfn{Cammino}{\label{def:cammino}
  Un \textit{cammino} di lunghezza $ \ell $ da $ u $ a $ v $ in $ G $ e' una sequenza
  \[
    v_1\, e_1\, v_2\, e_2\, \cdots\, e_\ell\, v_{\ell+1}
  \]
  con $ v_1 = u $, $ v_{\ell+1} = v $ e $ \varphi(e_i) = v_i v_{i+1} $ per ogni $ i $.
}

\nt{
  Invertire l'ordine di un cammino da' un cammino fra gli stessi due vertici, di stessa lunghezza.
}

\thm{Potenze della matrice di adiacenza}{\label{thm:potenze-A}
  Per ogni $ \ell \in \mathbb{Z}_{\geq 1} $:
  \[
    \bigl(A(G)^\ell\bigr)_{ij} = \#\{\text{cammini di lunghezza } \ell \text{ da } v_i \text{ a } v_j\} .
  \]
}
\nt{
  E' lo stesso principio delle \textit{catene di Markov}: le potenze della matrice di transizione contano le traiettorie di lunghezza fissata.
}
\pf{Dim}{
  Conseguenza diretta della definizione di prodotto di matrici. Posto $ A := A(G) $ con entrate $ a_{rs} $, e detto $ p = |V| $ il numero di vertici di $ G $:
  \[
    \bigl(A^\ell\bigr)_{ij}
      = \sum_{i_1, \ldots, i_{\ell-1} = 1}^{p} a_{i, i_1}\, a_{i_1, i_2}\, \cdots\, a_{i_{\ell-1}, j}\,.
  \]
  Fissata la sequenza di indici intermedi $ (i_1, \ldots, i_{\ell-1}) $, il prodotto $ a_{i, i_1} \cdots a_{i_{\ell-1}, j} $ vale il numero di cammini di lunghezza $ \ell $ da $ v_i $ a $ v_j $ che passano \textbf{esattamente} per la sequenza di vertici $ v_i, v_{i_1}, \ldots, v_{i_{\ell-1}}, v_j $: ci sono $ a_{i, i_1} $ scelte per il primo lato, $ a_{i_1, i_2} $ per il secondo, e cosi' via.

  Sommando su tutte le sequenze di vertici intermedi si ottengono tutti i cammini di lunghezza $ \ell $ da $ v_i $ a $ v_j $.

  (Per $ \ell = 1 $ non ci sono indici intermedi: $ (A^1)_{ij} = a_{ij} $ e' il numero di lati fra $ v_i $ e $ v_j $, ossia il numero di cammini di lunghezza $ 1 $.)
}

\subsubsection{Formula spettrale}

\nt{\textbf{Ripasso (Geometria 1).} Per il teorema spettrale $ A(G) $, essendo simmetrica reale, ammette una base di autovettori $ u_1, \ldots, u_p $ \textbf{ortonormali} ($ u_i^T u_j = \delta_{ij} $). Posto $ U := (u_1 \mid \cdots \mid u_p) $:
\[
  U^T = U^{-1}, \qquad U^T A(G)\, U = D := \mathrm{diag}(\lambda_1, \ldots, \lambda_p) ,
\]
con $ \lambda_i $ autovalore di $ u_i $.}

\cor{Formula spettrale per $ (A(G)^\ell)_{ij} $}{\label{cor:formula-spettrale}
  Per ogni coppia di vertici $ v_i, v_j $ e ogni $ \ell \in \mathbb{Z}_{\geq 1} $:
  \[
    \bigl(A(G)^\ell\bigr)_{ij} = c_1\, \lambda_1^\ell + \cdots + c_p\, \lambda_p^\ell ,
  \]
  con $ c_k := u_{ik}\, u_{jk} \in \mathbb{R} $ (componenti $ i $-esima e $ j $-esima dell'autovettore $ u_k $).
}
\pf{Dim}{
  $ A(G) $ e' simmetrica reale, quindi per il teorema spettrale ammette $ p $ autovettori reali $ u_1, \ldots, u_p $ linearmente indipendenti, scelti ortonormali. Sia $ U = (u_1 \mid \cdots \mid u_p) $ la matrice con tali autovettori per colonne: $ U $ e' \textbf{ortogonale}, ossia $ U^{-1} = U^T $, e diagonalizza $ A(G) $:
  \[
    U^{-1}\, A(G)\, U = \mathrm{diag}(\lambda_1, \ldots, \lambda_p)\,.
  \]
  Elevando a potenza $ \ell $ entrambi i membri (i termini $ U U^{-1} $ centrali si semplificano):
  \[
    U^{-1}\, A(G)^\ell\, U = \mathrm{diag}(\lambda_1^\ell, \ldots, \lambda_p^\ell)\,.
  \]
  Moltiplicando a sinistra per $ U $ e a destra per $ U^{-1} $ si isola $ A(G)^\ell $; usando $ U^{-1} = U^T $:
  \[
    A(G)^\ell = U \cdot \mathrm{diag}(\lambda_1^\ell, \ldots, \lambda_p^\ell) \cdot U^T\,.
  \]
  L'entrata $ (i,j) $ di $ A(G)^\ell $ si ottiene moltiplicando l'entrata $ i $-esima della colonna $ k $ di $ U $ (cioe' $ u_{ik} $) per l'autovalore $ \lambda_k^\ell $ per l'entrata $ j $-esima della riga $ k $ di $ U^T $ (cioe' $ u_{jk} $), e sommando su $ k $:
  \[
    \bigl(A(G)^\ell\bigr)_{ij} = \sum_{k=1}^{p} u_{ik}\, \lambda_k^\ell\, u_{jk}\,.
  \]
  Ponendo $ c_k := u_{ik}\, u_{jk} $ si ha la tesi.
}

\subsubsection{Cammini chiusi e traccia}

\dfn{Cammino chiuso}{\label{def:cammino-chiuso}
  Un cammino e' \textit{chiuso} se il vertice finale coincide con quello iniziale.
}

\cor{Numero di cammini chiusi}{\label{cor:cammini-chiusi-traccia}
  Il numero totale di cammini chiusi di lunghezza $ \ell $ in $ G $ e'
  \[
    f_G(\ell) := \sum_{i=1}^{p} \bigl(A(G)^\ell\bigr)_{ii} = \mathrm{Tr}\bigl(A(G)^\ell\bigr) = \sum_{k=1}^{p} \lambda_k^\ell .
  \]
}
\pf{Dim}{
  Per il Teorema~\ref{thm:potenze-A}, $ \bigl(A(G)^\ell\bigr)_{ii} $ e' il numero di cammini chiusi di lunghezza $ \ell $ basati in $ v_i $. Sommando sulle posizioni iniziali otteniamo la traccia. L'uguaglianza $ \mathrm{Tr}(A^\ell) = \sum_k \lambda_k^\ell $ segue dall'invarianza della traccia per coniugio: $ \mathrm{Tr}(A^\ell) = \mathrm{Tr}(U D^\ell U^T) = \mathrm{Tr}(D^\ell) $.
}

\subsection{Il grafo completo $ K_p $}

\dfn{Grafo completo $ K_p $}{\label{def:Kp}
  $ K_p $ e' il grafo \textit{semplice} con $ p $ vertici e un lato per ogni coppia di vertici distinti, dunque $ \binom{p}{2} $ lati totali. La sua matrice di adiacenza e'
  \[
    A(K_p) = J - I ,
  \]
  dove $ J $ e' la matrice $ p \times p $ con tutte le entrate uguali a $ 1 $ e $ I $ l'identita'.
}

\ex{$ K_3 $}{
  Il grafo completo su $ 3 $ vertici e' un triangolo: $ \binom{3}{2} = 3 $ lati.
  \begin{center}
    \begin{tikzpicture}[
      vertex/.style={circle, fill=black, inner sep=1.8pt},
      every label/.style={inner sep=2pt},
      thick
    ]
      \node[vertex, label=above:$v_1$] (v1) at (90:1.4)  {};
      \node[vertex, label={[label distance=1pt]below left:$v_2$}]  (v2) at (210:1.4) {};
      \node[vertex, label={[label distance=1pt]below right:$v_3$}] (v3) at (330:1.4) {};
      \draw (v1) -- (v2);
      \draw (v2) -- (v3);
      \draw (v3) -- (v1);
    \end{tikzpicture}
  \end{center}
  Matrice di adiacenza:
  \[
    A(K_3) = J - I = \begin{pmatrix} 0 & 1 & 1 \\ 1 & 0 & 1 \\ 1 & 1 & 0 \end{pmatrix} .
  \]
  Autovalori: $ 2 $ (semplice, autovettore $ (1,1,1)^T $) e $ -1 $ (con molteplicita' $ 2 $).
}

\mlenma{Autovalori di $ J $}{\label{lem:autovalori-J}
  Gli autovalori della matrice $ J \in \mathbb{R}^{p \times p} $ sono:
  \begin{itemize}
    \item $ p $ con molteplicita' $ 1 $ (autovettore $ (1, 1, \ldots, 1)^T $);
    \item $ 0 $ con molteplicita' $ p - 1 $ (autospazio: vettori a somma nulla).
  \end{itemize}
}

\mprop{Autovalori di $ A(K_p) $}{\label{prop:autovalori-Kp}
  Gli autovalori di $ A(K_p) $ sono:
  \begin{itemize}
    \item $ p - 1 $ con molteplicita' $ 1 $;
    \item $ -1 $ con molteplicita' $ p - 1 $.
  \end{itemize}
}
\pf{Dim}{
  $ v \neq 0 $ e' autovettore di $ A(K_p) = J - I $ con autovalore $ \lambda $ sse $ (J - I)v = \lambda v $, sse $ Jv = (\lambda + 1) v $. Quindi gli autovalori di $ A(K_p) $ sono quelli di $ J $ (Lemma~\ref{lem:autovalori-J}) shifted di $ -1 $: $ p \mapsto p - 1 $ e $ 0 \mapsto -1 $, con le stesse molteplicita'.
}

\cor{Cammini chiusi in $ K_p $ basati in un vertice}{\label{cor:cammini-chiusi-Kp}
  Per ogni $ \ell \geq 1 $ e ogni vertice $ v_i $ di $ K_p $:
  \[
    \bigl(A(K_p)^\ell\bigr)_{ii} = \frac{1}{p}\Bigl( (p-1)^\ell + (p-1)\,(-1)^\ell \Bigr) .
  \]
}
\pf{Dim}{
  Per il Corollario~\ref{cor:cammini-chiusi-traccia} il numero totale di cammini chiusi di lunghezza $ \ell $ in $ K_p $ e'
  \[
    f_{K_p}(\ell) = (p-1)^\ell + \underbrace{(-1)^\ell + \cdots + (-1)^\ell}_{p - 1 \text{ volte}} = (p-1)^\ell + (p-1)\,(-1)^\ell .
  \]
  Per simmetria del grafo completo, ogni vertice contribuisce con lo stesso numero di cammini chiusi: basta dividere per $ p $.
}

\osservazione{
  $ \bigl(A(K_p)^\ell\bigr)_{ii} $ coincide con il numero di successioni $ (i_1, i_2, \ldots, i_{\ell+1}) $ con $ i_j \in \{1, \ldots, p\} $, $ i_1 $ fissato, $ i_{j+1} \neq i_j $ per ogni $ j $ e $ i_{\ell+1} = i_1 $: ad ogni passo, su $ K_p $, si puo' andare a uno qualsiasi degli altri $ p - 1 $ vertici.
}

\subsubsection{Cammini non chiusi in $ K_p $: due metodi}

Quanti sono i cammini di lunghezza $ \ell $ in $ K_p $ da $ v_i $ a $ v_j $ con $ j \neq i $?

\paragraph{Primo metodo: totale meno chiusi.}

I cammini di lunghezza $ \ell $ che partono da $ v_i $ sono $ (p-1)^\ell $: ad ogni passo si hanno $ p - 1 $ scelte (uno qualsiasi degli altri vertici). Sottraendo i chiusi (Corollario~\ref{cor:cammini-chiusi-Kp}) si ottiene il numero di cammini non chiusi che partono da $ v_i $, ripartiti uniformemente fra i $ p - 1 $ vertici di arrivo $ v_j \neq v_i $:
\[
  \bigl(A(K_p)^\ell\bigr)_{ij}
    = \frac{1}{p-1}\!\left( (p-1)^\ell - \frac{(p-1)^\ell + (p-1)(-1)^\ell}{p} \right)
    = \frac{(p-1)^\ell - (-1)^\ell}{p} .
\]

\paragraph{Secondo metodo: espansione binomiale di $ (J - I)^\ell $.}

Sviluppiamo $ (J - I)^\ell $ con il binomio di Newton, valido perche' $ I $ e $ J $ commutano:
\[
  (J - I)^\ell = \sum_{k=0}^{\ell} (-1)^{\ell - k} \binom{\ell}{k}\, J^k .
\]

\osservazione{$ J^k = p^{k-1}\, J $ per ogni $ k \geq 1 $. Infatti $ J^2 $ ha entrata $ (i,j) $ uguale a $ \sum_h 1 \cdot 1 = p $, quindi $ J^2 = pJ $; per induzione, $ J^{k+1} = J \cdot J^k = J \cdot p^{k-1} J = p^{k-1}\, J^2 = p^k J $.}

Sostituendo (e isolando il termine $ k = 0 $, cioe' $ J^0 = I $):
\[
  (J - I)^\ell
    = (-1)^\ell I + \sum_{k=1}^{\ell} (-1)^{\ell - k} \binom{\ell}{k}\, p^{k-1}\, J
    = (-1)^\ell I + \frac{1}{p}\!\left( \sum_{k=1}^{\ell} (-1)^{\ell - k} \binom{\ell}{k}\, p^k \right)\! J .
\]
La somma fra parentesi e' $ (p - 1)^\ell - (-1)^\ell $ (binomio di Newton applicato a $ p $ e $ -1 $, con il termine $ k = 0 $ tolto), quindi:
\[
  \boxed{\;(J - I)^\ell = (-1)^\ell\, I + \frac{(p-1)^\ell - (-1)^\ell}{p}\, J\;}
\]
L'entrata $ (i,j) $ vale dunque:
\begin{itemize}
  \item per $ i = j $: $ (-1)^\ell + \frac{(p-1)^\ell - (-1)^\ell}{p} = \frac{(p-1)^\ell + (p-1)(-1)^\ell}{p} $ (coerente con il Corollario~\ref{cor:cammini-chiusi-Kp});
  \item per $ i \neq j $: $ \frac{(p-1)^\ell - (-1)^\ell}{p} $ (coerente con il primo metodo).
\end{itemize}

\osservazione{
  Il secondo metodo e' notevole perche' fornisce la formula chiusa per $ f_G(\ell) $ \textbf{senza conoscere gli autovalori}: si lavora puramente nella sottoalgebra commutativa generata da $ I $ e $ J $.
}

\subsection{Lemma di unicita' per somme di potenze}

\mlenma{Unicita' delle somme di potenze}{\label{lem:somma-potenze-unicita}
  Siano $ \alpha_1, \ldots, \alpha_r, \beta_1, \ldots, \beta_s \in \mathbb{C} \setminus \{0\} $ tali che per ogni $ \ell \in \mathbb{Z}_{\geq 1} $:
  \[
    \alpha_1^\ell + \cdots + \alpha_r^\ell = \beta_1^\ell + \cdots + \beta_s^\ell .
  \]
  Allora $ r = s $ e, a meno di riordinamenti, $ \alpha_i = \beta_i $ per ogni $ i $.
}
\pf{Dim}{
  Moltiplichiamo entrambi i membri per $ x^\ell $ e sommiamo su $ \ell \geq 1 $. A sinistra:
  \[
    \sum_{\ell \geq 1} \bigl(\alpha_1^\ell + \cdots + \alpha_r^\ell\bigr) x^\ell
      = \sum_{i=1}^{r} \sum_{\ell \geq 1} (\alpha_i x)^\ell
      = \sum_{i=1}^{r} \frac{\alpha_i x}{1 - \alpha_i x}\,,
  \]
  usando la serie geometrica $ \sum_{\ell \geq 1} (\gamma x)^\ell = \gamma x / (1 - \gamma x) $. Analogamente a destra. Si ottiene l'identita' di funzioni razionali:
  \[
    \sum_{i=1}^{r} \frac{\alpha_i x}{1 - \alpha_i x} = \sum_{j=1}^{s} \frac{\beta_j x}{1 - \beta_j x}\,. \tag{$\ast$}
  \]
  Sia $ \{\delta_1, \ldots, \delta_q\} $ l'insieme dei valori \textbf{distinti} che compaiono in $ \{\alpha_1, \ldots, \alpha_r\} \cup \{\beta_1, \ldots, \beta_s\} $, e poniamo
  \[
    m_k := \#\{i \mid \alpha_i = \delta_k\}\,, \qquad m'_k := \#\{j \mid \beta_j = \delta_k\}\,.
  \]
  La tesi e' equivalente a $ m_k = m'_k $ per ogni $ k $. Riscrivendo $ (\ast) $:
  \[
    \sum_{k=1}^{q} (m_k - m'_k)\, \frac{\delta_k x}{1 - \delta_k x} = 0\,.
  \]
  Moltiplicando per $ 1 - \delta_k x $ e valutando in $ x = 1/\delta_k $, tutti gli addendi con indice $ \neq k $ si annullano (contengono il fattore $ 1 - \delta_k x $), mentre l'addendo $ k $-esimo da':
  \[
    (m_k - m'_k)\, \delta_k \cdot \tfrac{1}{\delta_k}\, \prod_{j \neq k}\!\left(1 - \tfrac{\delta_j}{\delta_k}\right) = 0\,.
  \]
  Poiche' $ \delta_j \neq \delta_k $ per $ j \neq k $, il produttoria e' non nulla, dunque $ m_k = m'_k $.
}


\section{Spettro di grafi bipartiti}

\dfn{Grafo bipartito}{\label{def:bipartito}
  Un grafo $ G = (V, E, \varphi) $ e' \textit{bipartito} se esiste una partizione $ V = A \sqcup B $ tale che ogni lato di $ G $ ha un estremo in $ A $ e l'altro in $ B $.
}

\ex{$ P_3 $ e' bipartito}{
  Il cammino $ P_3 $ con vertici $ \{v_1, v_2, v_3\} $ e lati $ v_1 v_2, v_2 v_3 $ e' bipartito con $ A = \{v_1, v_3\} $, $ B = \{v_2\} $.
  \begin{center}
    \begin{tikzpicture}[
      vertex/.style={circle, fill=black, inner sep=1.8pt},
      every label/.style={inner sep=2pt},
      thick
    ]
      \node[vertex, label=below:$v_1$] (v1) at (-1.5, 0) {};
      \node[vertex, label=below:$v_2$] (v2) at (0, 0)    {};
      \node[vertex, label=below:$v_3$] (v3) at (1.5, 0)  {};
      \draw (v1) -- (v2) -- (v3);
    \end{tikzpicture}
  \end{center}
  Matrice di adiacenza:
  \[
    A(P_3) = \begin{pmatrix} 0 & 1 & 0 \\ 1 & 0 & 1 \\ 0 & 1 & 0 \end{pmatrix}\,,
  \]
  con autovalori $ \sqrt{2}, 0, -\sqrt{2} $. Osserviamo che $ 0 $ e' autovalore e che lo spettro e' \textbf{simmetrico} rispetto a $ 0 $.
}

\nt{In un grafo bipartito, ogni cammino chiuso ha lunghezza \textbf{pari}: ogni lato attraversa la partizione, quindi un cammino chiuso che ritorna al punto di partenza deve fare un numero pari di attraversamenti. In particolare $ f_G(\ell) = 0 $ per ogni $ \ell $ dispari.}

\mprop{Spettro di un grafo bipartito}{\label{prop:spettro-bipartito}
  Se $ G $ e' bipartito e $ \lambda $ e' autovalore di $ A(G) $, allora anche $ -\lambda $ e' autovalore (con stessa molteplicita').
}

Prima di dimostrarla, richiamiamo lo strumento algebrico chiave.

\dfnc{Funzioni simmetriche elementari e potenze}{\label{def:e-p}
  Per variabili $ x_1, \ldots, x_p $:
  \[
    e_k := \sum_{1 \leq i_1 < i_2 < \cdots < i_k \leq p} x_{i_1} x_{i_2} \cdots x_{i_k}\,, \qquad p_\ell := \sum_{i=1}^{p} x_i^\ell\,,
  \]
  con la convenzione $ e_0 = 1 $.
}

\thm{Formula di Newton}{\label{thm:newton}
  Per ogni $ k \geq 1 $:
  \[
    k\, e_k = \sum_{i=1}^{k} (-1)^{i-1}\, p_i\, e_{k-i}\,.
  \]
}

\pf{Dim della Proposizione~\ref{prop:spettro-bipartito}}{
  Siano $ \lambda_1, \ldots, \lambda_p $ gli autovalori di $ A(G) $ (con molteplicita') e $ \tilde{e}_k := e_k(\lambda_1, \ldots, \lambda_p) $. Il polinomio caratteristico e':
  \[
    p_{A(G)}(x) = \det(x I - A(G)) = \prod_{i=1}^{p} (x - \lambda_i) = x^p - \tilde{e}_1 x^{p-1} + \tilde{e}_2 x^{p-2} - \cdots + (-1)^p \tilde{e}_p\,.
  \]
  Dalla traccia delle potenze, $ p_\ell(\lambda_1, \ldots, \lambda_p) = \sum_i \lambda_i^\ell = \mathrm{Tr}(A(G)^\ell) = f_G(\ell) $. Per la nota sopra, $ G $ bipartito $ \Longrightarrow p_\ell = 0 $ per ogni $ \ell $ dispari.

  \textbf{Per induzione su $ k $ dispari mostriamo $ \tilde{e}_k = 0 $.}
  \begin{itemize}
    \item \textbf{Base ($ k = 1 $):} $ \tilde{e}_1 = p_1 = 0 $.
    \item \textbf{Passo:} sia $ k \geq 3 $ dispari, e supponiamo $ \tilde{e}_1 = \tilde{e}_3 = \cdots = \tilde{e}_{k-2} = 0 $. Per Newton:
    \[
      k\, \tilde{e}_k = \sum_{i=1}^{k} (-1)^{i-1}\, p_i\, \tilde{e}_{k-i}\,.
    \]
    Per ogni $ i $:
    \begin{itemize}
      \item se $ i $ e' dispari, $ p_i = 0 $ (cammini chiusi di lunghezza dispari);
      \item se $ i $ e' pari, $ k - i $ e' dispari (perche' $ k $ dispari), quindi $ \tilde{e}_{k-i} = 0 $ per ipotesi induttiva.
    \end{itemize}
    Ogni addendo si annulla, dunque $ \tilde{e}_k = 0 $.
  \end{itemize}
  Quindi $ \tilde{e}_k = 0 $ per ogni $ k $ dispari, e il polinomio caratteristico contiene solo monomi con la stessa parita' di $ p $:
  \[
    p_{A(G)}(x) = x^p + \tilde{e}_2\, x^{p-2} + \tilde{e}_4\, x^{p-4} + \cdots\,.
  \]
  Sostituendo $ x \mapsto -x $: $ (-x)^{p - 2j} = (-1)^p\, x^{p - 2j} $ per ogni $ j $, da cui
  \[
    p_{A(G)}(-x) = (-1)^p\, p_{A(G)}(x)\,.
  \]
  In particolare $ p_{A(G)}(\lambda) = 0 \Longrightarrow p_{A(G)}(-\lambda) = 0 $: $ -\lambda $ e' autovalore con la stessa molteplicita'.
}

\ex{Esercizio: $ P_n $ con $ n $ dispari}{
  Sia $ P_n $ il cammino con vertici $ v_1, v_2, \ldots, v_n $ e lati $ v_i v_{i+1} $. $ P_n $ e' bipartito (vertici di indice pari/dispari). Per $ n $ dispari, lo spettro e' simmetrico rispetto a $ 0 $: ci sono $ n $ autovalori con $ \lambda \leftrightarrow -\lambda $, ma $ n $ dispari forza l'esistenza di $ \lambda = 0 $ (l'unico fissato dall'involuzione).
}

\subsection{Il grafo bipartito completo $ K_{r,s} $}

\dfn{Grafo bipartito completo $ K_{r,s} $}{\label{def:Krs}
  $ K_{r,s} $ e' il grafo bipartito con partizione $ V = V_1 \sqcup V_2 $, $ |V_1| = r $, $ |V_2| = s $, e un lato fra \textbf{ogni} vertice di $ V_1 $ e \textbf{ogni} vertice di $ V_2 $ (in tutto $ rs $ lati, nessuno interno a $ V_1 $ o $ V_2 $).
  \begin{center}
    \begin{tikzpicture}[
      vertex/.style={circle, fill=black, inner sep=1.6pt},
      thick, scale=0.9
    ]
      \foreach \i in {1,2,3} \node[vertex] (a\i) at (0, {2-\i}) {};
      \foreach \j in {1,2} \node[vertex] (b\j) at (2.4, {1.5-\j}) {};
      \foreach \i in {1,2,3} \foreach \j in {1,2} \draw (a\i) -- (b\j);
      \node at (-0.6, 0) {$ V_1 $};
      \node at (3.0, 0) {$ V_2 $};
    \end{tikzpicture}
  \end{center}
  La matrice di adiacenza, ordinando prima i vertici di $ V_1 $ e poi quelli di $ V_2 $, ha la struttura a blocchi
  \[
    A(K_{r,s}) = \begin{pmatrix} 0_{r \times r} & J_{r \times s} \\ J_{s \times r} & 0_{s \times s} \end{pmatrix},
  \]
  dove $ J $ denota blocchi di soli $ 1 $. Ha \textbf{rango $ 2 $}, dunque $ 0 $ e' autovalore con molteplicita' $ r + s - 2 $.
}

\mprop{Spettro di $ K_{r,s} $}{\label{prop:spettro-Krs}
  Gli autovalori di $ A(K_{r,s}) $ sono:
  \begin{itemize}
    \item $ +\sqrt{rs} $ e $ -\sqrt{rs} $, ciascuno con molteplicita' $ 1 $;
    \item $ 0 $ con molteplicita' $ r + s - 2 $.
  \end{itemize}
}
\pf{Dim}{
  \textbf{Conteggio dei cammini chiusi.} Per la nota sulla bipartizione $ f_{K_{r,s}}(\ell) = 0 $ se $ \ell $ e' dispari. Per $ \ell = 2m $ pari contiamo $ f_{K_{r,s}}(2m) = \mathrm{Tr}(A^{2m}) $. Osserviamo che
  \[
    A^2 = \begin{pmatrix} J_{r \times s} J_{s \times r} & 0 \\ 0 & J_{s \times r} J_{r \times s} \end{pmatrix} = \begin{pmatrix} s\, J_{r \times r} & 0 \\ 0 & r\, J_{s \times s} \end{pmatrix},
  \]
  perche' $ J_{r \times s} J_{s \times r} = s\, J_{r \times r} $ (ogni entrata e' $ \sum_{k=1}^{s} 1 = s $). Usando $ J_{r \times r}^m = r^{m-1} J_{r \times r} $:
  \[
    A^{2m} = \begin{pmatrix} s^m r^{m-1}\, J_{r \times r} & 0 \\ 0 & r^m s^{m-1}\, J_{s \times s} \end{pmatrix}.
  \]
  La traccia: il blocco superiore contribuisce $ r \cdot s^m r^{m-1} = (rs)^m $, quello inferiore $ s \cdot r^m s^{m-1} = (rs)^m $, quindi
  \[
    f_{K_{r,s}}(2m) = 2\,(rs)^m, \qquad \text{ossia} \qquad f_{K_{r,s}}(\ell) = \begin{cases} 2\,(rs)^{\ell/2} & \ell \text{ pari}, \\ 0 & \ell \text{ dispari}. \end{cases}
  \]

  \textbf{Estrazione degli autovalori.} Detti $ \lambda_1, \ldots, \lambda_{r+s} $ gli autovalori, $ \sum_i \lambda_i^\ell = f_{K_{r,s}}(\ell) $. Poiche' $ A(K_{r,s}) $ ha rango $ 2 $, almeno $ r + s - 2 $ autovalori sono nulli; siano $ \lambda_1, \lambda_2 $ i (al piu') due non nulli. Allora per ogni $ \ell $:
  \[
    \lambda_1^\ell + \lambda_2^\ell = 2\,(rs)^{\ell/2} = (\sqrt{rs})^\ell + (-\sqrt{rs})^\ell
  \]
  (l'ultima uguaglianza vale perche' per $ \ell $ dispari entrambi i membri sono nulli, per $ \ell $ pari entrambi valgono $ 2(rs)^{\ell/2} $). Per il Lemma di unicita' delle somme di potenze (Lemma~\ref{lem:somma-potenze-unicita}), $ \{\lambda_1, \lambda_2\} = \{\sqrt{rs}, -\sqrt{rs}\} $. Lo spettro e' dunque $ \pm\sqrt{rs} $ (semplici) e $ 0 $ con molteplicita' $ r + s - 2 $ --- coerente con la simmetria dei grafi bipartiti (Proposizione~\ref{prop:spettro-bipartito}).
}


\section{Il cubo $ n $-dimensionale $ C_n $}

\subsection{Definizione}

\dfn{Cubo $ n $-dimensionale}{\label{def:cubo}
  Sia $ \mathbb{Z}_2 = \{0, 1\} $ con somma modulo $ 2 $. Il \textit{cubo $ n $-dimensionale} (o \textit{ipercubo}) $ C_n $ e' il grafo con:
  \begin{itemize}
    \item vertici $ V(C_n) = \mathbb{Z}_2^n = \underbrace{\mathbb{Z}_2 \times \cdots \times \mathbb{Z}_2}_{n} $;
    \item un lato fra $ u $ e $ v $ se differiscono in \textbf{esattamente una componente}.
  \end{itemize}
  $ C_n $ ha $ 2^n $ vertici e ogni vertice ha grado $ n $.
}

\dfn{Peso di Hamming}{\label{def:hamming}
  Per $ u \in C_n $ si pone
  \[
    \omega(u) := \#\{i \in \{1, \ldots, n\} \mid u_i = 1\}\,,
  \]
  detto \textit{peso di Hamming} di $ u $. Allora c'e' un lato fra $ u $ e $ v $ in $ C_n $ sse $ \omega(u + v) = 1 $.
}

\subsection{Caratteri di $ C_n $}

\dfn{Prodotto scalare in $ \mathbb{Z}_2^n $}{\label{def:prod-Z2}
  Per $ u = (u_1, \ldots, u_n) $ e $ v = (v_1, \ldots, v_n) \in \mathbb{Z}_2^n $:
  \[
    u \cdot v := u_1 v_1 + u_2 v_2 + \cdots + u_n v_n \pmod 2 \in \{0, 1\}\,.
  \]
}

\dfn{Caratteri di $ C_n $}{\label{def:caratteri-cubo}
  Per $ u \in C_n $ definiamo il \textit{carattere} $ \chi_u : C_n \to \mathbb{R} $ come
  \[
    \chi_u(v) := (-1)^{u \cdot v} = \begin{cases} +1 & \text{se } u \cdot v = 0\,, \\ -1 & \text{se } u \cdot v = 1\,. \end{cases}
  \]
}

\mprop{$ \chi_u $ e' un omomorfismo}{\label{prop:caratteri-omom}
  Per ogni $ u \in C_n $, il carattere $ \chi_u $ e' un \textbf{omomorfismo di gruppi} $ (C_n, +) \to (\mathbb{R}^\times, \cdot) $, cioe' per ogni $ v, w \in C_n $:
  \[
    \chi_u(v + w) = \chi_u(v)\, \chi_u(w)\,.
  \]
}
\pf{Dim}{
  Per ogni $ v, w \in C_n $:
  \[
    \chi_u(v + w) = (-1)^{u \cdot (v + w)} = (-1)^{u \cdot v + u \cdot w} = (-1)^{u \cdot v} \cdot (-1)^{u \cdot w} = \chi_u(v)\, \chi_u(w)\,.
  \]
}
\osservazione{
  I $ \chi_u $ con $ u \in C_n $ sono i $ 2^n $ caratteri irriducibili distinti di $ C_n $, dunque sono \textbf{tutti} i caratteri di $ C_n $.
}

\subsection{Spazio delle funzioni e trasformata di Radon}

\dfn{Spazio $ V $ e base canonica}{\label{def:spazio-V-cubo}
  $ V := \{g : C_n \to \mathbb{R}\} $ e' uno spazio vettoriale reale di dimensione $ |C_n| = 2^n $.

  La \textit{base canonica} $ \{\delta_u\}_{u \in C_n} $ e' data da
  \[
    \delta_u(v) := \delta_{u, v} = \begin{cases} 1 & \text{se } u = v\,, \\ 0 & \text{se } u \neq v\,. \end{cases}
  \]
  Ogni $ g \in V $ si scrive in modo unico come $ g = \sum_{u \in C_n} g(u)\, \delta_u $.
}

\dfn{Trasformata di Radon discreta}{\label{def:radon}
  Sia $ P \subseteq C_n $. La \textit{trasformata di Radon discreta} associata a $ P $ e' l'operatore lineare $ \Phi_P : V \to V $ definito da
  \[
    \Phi_P(g)(v) := \sum_{w \in P} g(v + w)\,.
  \]
}

\thm{Diagonalizzazione di $ \Phi_P $}{\label{thm:radon-diag}
  Gli autovettori di $ \Phi_P $ sono i caratteri $ \chi_u $, e l'autovalore associato a $ \chi_u $ e'
  \[
    \lambda_u = \sum_{w \in P} (-1)^{u \cdot w} = \sum_{w \in P} \chi_u(w)\,.
  \]
  La base $ \{\chi_u \mid u \in C_n\} $ diagonalizza \textbf{ogni} $ \Phi_P $.
}

\pf{Dim}{
  Per ogni $ v \in C_n $:
  \[
    \Phi_P(\chi_u)(v) = \sum_{w \in P} \chi_u(v + w) \stackrel{\text{Prop.~\ref{prop:caratteri-omom}}}{=} \sum_{w \in P} \chi_u(v)\, \chi_u(w) = \chi_u(v) \cdot \underbrace{\sum_{w \in P} \chi_u(w)}_{= \lambda_u}\,.
  \]
  Dunque $ \Phi_P(\chi_u) = \lambda_u\, \chi_u $. I $ 2^n $ caratteri $ \chi_u $ sono ortogonali (caratteri irriducibili distinti), dunque linearmente indipendenti, e formano una base di $ V $ che diagonalizza $ \Phi_P $.
}

\nt{
  Quando $ P $ e' l'insieme dei vettori di peso $ 1 $ (cioe' la base canonica $ \{e_1, \ldots, e_n\} $ di $ \mathbb{Z}_2^n $), $ \Phi_P $ coincide con l'\textbf{operatore di adiacenza} di $ C_n $: $ \Phi_P(g)(v) = \sum_{w \sim v} g(w) $.
}

\cor{Autovettori e autovalori di $ A(C_n) $}{\label{cor:spettro-cubo}
  Sia $ P = \{e_1, \ldots, e_n\} $ l'insieme dei vettori di peso $ 1 $ in $ \mathbb{Z}_2^n $, cosicche' $ \Phi_P = A(C_n) $. Per ogni $ u \in \mathbb{Z}_2^n $:
  \begin{itemize}
    \item l'autovettore associato a $ u $, visto come combinazione lineare dei vertici di $ C_n $, e'
    \[
      E_u = \sum_{v \in \mathbb{Z}_2^n} (-1)^{u \cdot v}\, v\,;
    \]
    \item il corrispondente autovalore e'
    \[
      \lambda_u = n - 2\, \omega(u)\,,
    \]
    dove $ \omega(u) $ e' il peso di Hamming di $ u $.
  \end{itemize}
  In particolare $ A(C_n) $ ha autovalore $ n - 2k $ con molteplicita' $ \binom{n}{k} $, per $ k = 0, 1, \ldots, n $. Lo spettro e' simmetrico rispetto a $ 0 $ (\(k \leftrightarrow n - k\)), coerentemente con la Proposizione~\ref{prop:spettro-bipartito}.
}
\pf{Dim}{
  Per ogni $ g \in V $ vale la decomposizione $ g = \sum_v g(v)\, \delta_v $. Applicandola al carattere $ \chi_u $:
  \[
    \chi_u = \sum_{v \in \mathbb{Z}_2^n} \chi_u(v)\, \delta_v = \sum_{v \in \mathbb{Z}_2^n} (-1)^{u \cdot v}\, \delta_v\,.
  \]
  La matrice di $ \Phi_P $ rispetto alla base $ \{\delta_v\} $ coincide con $ A(C_n) $ (rispetto ai vertici $ v $): identificando $ \delta_v \leftrightarrow v $, l'autovettore $ \chi_u $ corrisponde a $ E_u = \sum_v (-1)^{u \cdot v}\, v $.

  Per l'autovalore, dal Teorema~\ref{thm:radon-diag}:
  \[
    \lambda_u = \sum_{w \in P} (-1)^{u \cdot w} = \sum_{i=1}^{n} (-1)^{u \cdot e_i} = \sum_{i=1}^{n} (-1)^{u_i}\,,
  \]
  poiche' $ e_i \cdot u = u_i $. Ci sono $ n - \omega(u) $ indici $ i $ con $ u_i = 0 $ (contributo $ +1 $) e $ \omega(u) $ indici con $ u_i = 1 $ (contributo $ -1 $), quindi
  \[
    \lambda_u = (n - \omega(u)) - \omega(u) = n - 2\, \omega(u)\,.
  \]
  Al variare di $ u \in \mathbb{Z}_2^n $, $ \omega(u) $ assume i valori $ 0, 1, \ldots, n $, e $ \#\{u \mid \omega(u) = k\} = \binom{n}{k} $.
}


\section{Operatori di convoluzione su gruppi abeliani finiti}

L'idea alla base della trasformata di Radon $ \Phi_P $ su $ C_n $ (Definizione~\ref{def:radon}) si generalizza a un qualsiasi gruppo abeliano finito $ G $: i caratteri di $ G $ giocano il ruolo dei $ \chi_u $ del cubo e diagonalizzano simultaneamente un'intera famiglia di operatori, gli \textit{operatori di convoluzione}. Il cubo $ C_n $ con la sua trasformata di Radon e' allora un caso particolare di questa costruzione, ottenuto per $ G = \mathbb{Z}_2^n $ e $ \Gamma = \{e_1, \ldots, e_n\} $.

\subsection{Operatori di traslazione e di convoluzione}

\dfn{Spazio delle funzioni e operatori di traslazione}{\label{def:traslazione}
  Sia $ G $ un gruppo abeliano finito (additivo). Lo spazio delle funzioni complesse su $ G $ e'
  \[
    V := \{g : G \to \mathbb{C}\}\,,
  \]
  uno spazio vettoriale complesso di dimensione $ |G| $. Per ogni $ a \in G $ definiamo l'\textit{operatore di traslazione} $ T_a : V \to V $ come
  \[
    T_a(g)(x) := g(x + a)\,.
  \]
}

\nt{
  $ T_a $ e' lineare. Per ogni $ a, b \in G $ vale $ T_a\, T_b = T_{a + b} $ (associativita' della somma in $ G $), e poiche' $ G $ e' \textbf{abeliano}:
  \[
    T_a\, T_b = T_{a + b} = T_{b + a} = T_b\, T_a\,.
  \]
  I $ T_a $ commutano fra loro: l'algebra che generano e' commutativa.
}

\dfn{Operatore di convoluzione}{\label{def:convoluzione}
  Dati coefficienti $ \{c_a\}_{a \in G} \subseteq \mathbb{C} $, l'\textit{operatore di convoluzione} associato e'
  \[
    A := \sum_{a \in G} c_a\, T_a \,:\, V \longrightarrow V\,.
  \]
  Caso particolare: fissato un sottoinsieme $ \Gamma \subseteq G $ e prendendo
  \[
    c_a = \begin{cases} 1 & \text{se } a \in \Gamma\,, \\ 0 & \text{altrimenti,} \end{cases}
  \]
  si recupera la trasformata di Radon discreta $ \Phi_\Gamma $ (Definizione~\ref{def:radon}, qui per un qualsiasi gruppo $ G $).
}

\osservazione{
  Il nome \textit{convoluzione} viene dalla formula $ A(g)(x) = \sum_{a \in G} c_a\, g(x + a) $, che e' la convoluzione discreta di $ g $ con la funzione $ a \mapsto c_a $.
}

\subsection{Caratteri di un gruppo abeliano finito}

\dfn{Carattere}{\label{def:carattere-generale}
  Un \textit{carattere} del gruppo abeliano finito $ G $ e' un omomorfismo di gruppi
  \[
    \chi : (G, +) \longrightarrow (\mathbb{C}^\times, \cdot)\,,
  \]
  ossia $ \chi(a + b) = \chi(a)\, \chi(b) $ per ogni $ a, b \in G $.
}

\nt{
  Poiche' ogni $ a \in G $ ha ordine finito, $ \chi(a) $ e' radice dell'unita': l'immagine di un carattere e' contenuta in $ S^1 = \{z \in \mathbb{C} : |z| = 1\} $. Inoltre, essendo $ G $ abeliano, tutti i caratteri irriducibili di $ G $ hanno dimensione $ 1 $ (sono cioe' i caratteri nel senso classico). Denotiamo
  \[
    \widehat{G} := \{\chi : G \to \mathbb{C}^\times \mid \chi \text{ omomorfismo}\}\,;
  \]
  $ \widehat{G} $ e' un gruppo abeliano (prodotto puntuale) e
  \[
    |\widehat{G}| = |G|\,,
  \]
  perche' $ |G| = \sum_{\chi \text{ irr.}} \chi(1)^2 $ e per $ G $ abeliano $ \chi(1) = 1 $ per ogni $ \chi $.
}

\ex{Caratteri di $ C_n = \mathbb{Z}_2^n $}{
  Per il cubo, i caratteri sono i $ \chi_u(v) = (-1)^{u \cdot v} $ con $ u \in \mathbb{Z}_2^n $ (Definizione~\ref{def:caratteri-cubo}): sono $ 2^n $ omomorfismi distinti, dunque tutti e soli i caratteri di $ \mathbb{Z}_2^n $. In questo caso particolare i valori sono in $ \{\pm 1\} \subset \mathbb{R} $, e non in $ \mathbb{C} $ propriamente.
}

\subsection{Diagonalizzazione simultanea}

\mprop{Caratteri come autovettori di $ T_a $}{\label{prop:Ta-diag}
  Per ogni $ a \in G $ e ogni carattere $ \chi $ di $ G $:
  \[
    T_a(\chi) = \chi(a)\, \chi\,.
  \]
  In particolare $ \chi $ e' autovettore di $ T_a $ con autovalore $ \chi(a) $.
}
\pf{Dim}{
  Per ogni $ x \in G $:
  \[
    T_a(\chi)(x) = \chi(x + a) = \chi(x)\, \chi(a) = \chi(a)\, \chi(x)\,,
  \]
  usando che $ \chi $ e' omomorfismo. Dunque $ T_a(\chi) = \chi(a)\, \chi $ come elementi di $ V $.
}

\thm{Diagonalizzazione simultanea (versione generale del Teorema~\ref{thm:radon-diag})}{\label{thm:diag-gen}
  Sia $ G $ abeliano finito e $ A = \sum_{a \in G} c_a\, T_a $ un operatore di convoluzione. Ogni carattere $ \chi $ di $ G $ e' autovettore di $ A $ con autovalore
  \[
    \lambda_\chi = \sum_{a \in G} c_a\, \chi(a)\,.
  \]
  I $ |G| $ caratteri di $ G $ formano una base di $ V $ che diagonalizza simultaneamente \textbf{tutti} gli operatori di convoluzione. In particolare, per la trasformata di Radon $ \Phi_\Gamma $ (caso $ c_a = \mathbb{1}_\Gamma(a) $):
  \[
    \lambda_\chi = \sum_{a \in \Gamma} \chi(a)\,.
  \]
}
\pf{Dim}{
  Per linearita' di $ A $ e per la Proposizione~\ref{prop:Ta-diag}:
  \[
    A(\chi) = \sum_{a \in G} c_a\, T_a(\chi) = \sum_{a \in G} c_a\, \chi(a)\, \chi = \Bigl(\sum_{a \in G} c_a\, \chi(a)\Bigr)\, \chi\,.
  \]
  I $ |G| = |\widehat{G}| $ caratteri di $ G $ sono linearmente indipendenti (relazioni di ortogonalita' dei caratteri) e $ \dim V = |G| $, dunque formano una base di $ V $.
}

\osservazione{
  Il Teorema~\ref{thm:radon-diag} per il cubo si recupera prendendo $ G = \mathbb{Z}_2^n $ e $ \Gamma = \{e_1, \ldots, e_n\} $: i caratteri sono i $ \chi_u(v) = (-1)^{u \cdot v} $ e l'autovalore associato e'
  \[
    \lambda_{\chi_u} = \sum_{w \in \Gamma} \chi_u(w) = \sum_{i=1}^{n} (-1)^{u_i} = n - 2\, \omega(u)\,,
  \]
  in accordo con il Corollario~\ref{cor:spettro-cubo}.
}

\subsection{Formula esplicita per i cammini nel cubo}

Combinando lo spettro di $ A(C_n) $ (Corollario~\ref{cor:spettro-cubo}) con la formula spettrale per le potenze (Corollario~\ref{cor:formula-spettrale}), otteniamo una formula chiusa per il numero di cammini fra due vertici qualsiasi del cubo.

\cor{Numero di cammini in $ C_n $}{\label{cor:cammini-cubo}
  Siano $ u, v \in \mathbb{Z}_2^n $ con $ \omega(u + v) = k $. Il numero di cammini di lunghezza $ \ell $ in $ C_n $ da $ u $ a $ v $ e':
  \[
    \bigl(A(C_n)^\ell\bigr)_{u, v}
      = \frac{1}{2^n} \sum_{i = 0}^{n} \sum_{j = 0}^{k} (-1)^j \binom{k}{j} \binom{n - k}{i - j}\, (n - 2 i)^\ell\,,
  \]
  con la convenzione $ \binom{n - k}{i - j} = 0 $ se $ j > i $ o $ i - j > n - k $.
}

\osservazione{
  La formula dipende da $ u, v $ solo attraverso $ k = \omega(u + v) $, ossia il numero di coordinate in cui $ u $ e $ v $ differiscono (\textit{distanza di Hamming}). Coerentemente: il cubo $ C_n $ e' vertex-transitive e il numero di cammini di lunghezza $ \ell $ fra due vertici dipende solo dalla loro distanza.
}

\pf{Dim}{
  \textbf{Strategia.} Applichiamo la formula spettrale (Corollario~\ref{cor:formula-spettrale}): se $ \{\widehat{E}_w\}_{w \in C_n} $ e' una base \textbf{ortonormale} di autovettori di $ A(C_n) $ con autovalori $ \{\mu_w\} $, allora
  \[
    \bigl(A(C_n)^\ell\bigr)_{u, v} = \sum_{w \in C_n} (\widehat{E}_w)_u\, (\widehat{E}_w)_v\, \mu_w^\ell\,,
  \]
  dove $ (\widehat{E}_w)_u $ e' la componente $ u $-esima di $ \widehat{E}_w $ (la coordinata sul vettore $ u $ della base canonica $ \{\delta_u\} $).

  \textbf{Ortonormalizzazione.} Dal Corollario~\ref{cor:spettro-cubo}, $ E_w = \sum_{v \in C_n} (-1)^{w \cdot v}\, v $ e' un autovettore di autovalore $ \mu_w = n - 2\, \omega(w) $. La sua norma al quadrato e'
  \[
    |E_w|^2 = \sum_{v \in C_n} \bigl((-1)^{w \cdot v}\bigr)^2 = \sum_{v \in C_n} 1 = 2^n\,,
  \]
  quindi $ \widehat{E}_w := \tfrac{1}{2^{n/2}}\, E_w $ ha norma $ 1 $, e
  \[
    (\widehat{E}_w)_u = \tfrac{1}{2^{n/2}}\, (-1)^{u \cdot w}\,.
  \]

  \textbf{Sostituzione nella formula spettrale.}
  \[
    \bigl(A(C_n)^\ell\bigr)_{u, v}
      = \sum_{w \in C_n} \tfrac{(-1)^{u \cdot w}}{2^{n/2}} \cdot \tfrac{(-1)^{v \cdot w}}{2^{n/2}} \cdot (n - 2\, \omega(w))^\ell
      = \frac{1}{2^n} \sum_{w \in C_n} (-1)^{(u + v) \cdot w}\, (n - 2\, \omega(w))^\ell\,.
  \]

  \textbf{Raggruppamento dei termini.} Indicizziamo i $ w \in C_n $ in base a:
  \begin{itemize}
    \item $ i := \omega(w) \in \{0, 1, \ldots, n\} $ — il peso di $ w $;
    \item $ j := \#\{r \mid w_r = 1 \text{ e } (u + v)_r = 1\} \in \{0, 1, \ldots, k\} $ — il numero di indici in cui $ w $ e $ u + v $ hanno entrambi un $ 1 $.
  \end{itemize}
  Allora $ (u + v) \cdot w \equiv j \pmod 2 $ (somma in $ \mathbb{Z}_2 $ delle coordinate in comune), dunque $ (-1)^{(u + v) \cdot w} = (-1)^j $. E i $ w $ con $ \omega(w) = i $ e tutti gli stessi valori di $ (n - 2i)^\ell $ contribuiscono tutti uguale, quindi possiamo raggruppare.

  Contiamo gli $ w $ con una coppia $ (i, j) $ fissata. Pensiamo a $ w $ come al sottoinsieme dei suoi indici uguali a $ 1 $: dobbiamo scegliere
  \begin{itemize}
    \item $ j $ indici fra i $ k $ dove $ u + v $ vale $ 1 $: $ \binom{k}{j} $ scelte;
    \item $ i - j $ indici fra gli $ n - k $ dove $ u + v $ vale $ 0 $: $ \binom{n - k}{i - j} $ scelte (zero se $ j > i $).
  \end{itemize}
  In totale $ \binom{k}{j}\binom{n - k}{i - j} $ vettori $ w $. Sommando:
  \[
    \bigl(A(C_n)^\ell\bigr)_{u, v}
      = \frac{1}{2^n} \sum_{i = 0}^{n} \sum_{j = 0}^{k} (-1)^j \binom{k}{j} \binom{n - k}{i - j}\, (n - 2 i)^\ell\,.
  \]
}

\osservazione{
  \textbf{Caso $ k = 0 $ (cammini chiusi):} $ u = v $, $ \omega(u + v) = 0 $, la somma su $ j $ collassa al solo $ j = 0 $:
  \[
    \bigl(A(C_n)^\ell\bigr)_{u, u} = \frac{1}{2^n} \sum_{i = 0}^{n} \binom{n}{i}\, (n - 2 i)^\ell\,,
  \]
  indipendente dal vertice $ u $ (come deve essere per simmetria). Sommando su $ u $ si recupera $ f_{C_n}(\ell) = \sum_{i=0}^{n} \binom{n}{i}\, (n - 2 i)^\ell $, in accordo con il Corollario~\ref{cor:cammini-chiusi-traccia}.
}

\subsection{Il ciclo come grafo di Cayley di $ \mathbb{Z}/n\mathbb{Z} $}

Sia il cubo $ C_n $ sia il ciclo nascono dalla stessa costruzione: un \textit{grafo di Cayley} su un gruppo abeliano, in cui la trasformata di Radon coincide con l'operatore di adiacenza e i caratteri ne diagonalizzano la matrice.

\dfn{Ciclo $ \mathrm{Cic}_n $}{\label{def:ciclo}
  Sia $ G = \mathbb{Z}/n\mathbb{Z} $. Il \textit{ciclo} su $ n $ vertici e' il grafo $ \mathrm{Cic}_n $ con:
  \begin{itemize}
    \item vertici $ V = \mathbb{Z}/n\mathbb{Z} = \{0, 1, \ldots, n-1\} $;
    \item un lato fra $ i $ e $ j $ sse $ i - j \in \{1, n-1\} = \{\pm 1\} \pmod n $.
  \end{itemize}
  E' il grafo di Cayley $ \mathrm{Cay}(\mathbb{Z}/n\mathbb{Z}, \Delta) $ con insieme di connessione $ \Delta = \{1, n-1\} $ (simmetrico). Ogni vertice ha grado $ 2 $; il grafo e' un poligono di $ n $ lati. (Da non confondere col cubo $ C_n $ del~\S3.)
}

\nt{
  Come per il cubo (Corollario~\ref{cor:spettro-cubo}), la matrice di $ \Phi_\Delta $ rispetto alla base $ \{\delta_u\}_{u \in \mathbb{Z}/n\mathbb{Z}} $ e' esattamente la matrice di adiacenza $ A(\mathrm{Cic}_n) $: infatti
  \[
    \Phi_\Delta(\delta_u)(x) = \delta_u(x + 1) + \delta_u(x - 1) = \delta_{u-1}(x) + \delta_{u+1}(x)\,,
  \]
  cioe' $ \Phi_\Delta(\delta_u) = \delta_{u-1} + \delta_{u+1} $: la colonna $ u $ ha un $ 1 $ in posizione $ u \pm 1 $, come $ A(\mathrm{Cic}_n) $. La matrice e' \textbf{circolante} (cfr.\ \S6.3).
}

\cor{Spettro del ciclo $ \mathrm{Cic}_n $}{\label{cor:spettro-ciclo}
  Sia $ \omega := e^{2\pi i / n} $. I caratteri di $ \mathbb{Z}/n\mathbb{Z} $ sono $ \chi_j(a) = \omega^{j a} $ ($ j = 0, \ldots, n-1 $). Gli autovettori di $ A(\mathrm{Cic}_n) $ sono i caratteri $ \chi_j $, con autovalore
  \[
    \lambda_j = \sum_{a \in \Delta} \chi_j(a) = \omega^{j} + \omega^{j(n-1)} = \omega^j + \omega^{-j} = 2\cos\!\Bigl(\frac{2\pi j}{n}\Bigr)\,, \qquad j = 0, 1, \ldots, n-1\,.
  \]
}
\pf{Dim}{
  Applichiamo il Teorema~\ref{thm:diag-gen} con $ G = \mathbb{Z}/n\mathbb{Z} $ e $ \Gamma = \Delta = \{1, n-1\} $: ogni carattere $ \chi_j $ e' autovettore di $ \Phi_\Delta = A(\mathrm{Cic}_n) $ con autovalore $ \sum_{a \in \Delta} \chi_j(a) $. Calcolando, $ \chi_j(1) = \omega^j $ e $ \chi_j(n-1) = \omega^{j(n-1)} = \omega^{-j} $ (poiche' $ \omega^{jn} = 1 $). Infine $ \omega^j + \omega^{-j} = 2\cos(2\pi j / n) $ per la formula di Eulero.
}

\osservazione{
  Lo spettro $ \{2\cos(2\pi j/n)\}_{j=0}^{n-1} $ e' reale (come deve, $ A $ simmetrica) e simmetrico rispetto a $ 0 $ solo quando $ n $ e' pari (i.e.\ quando $ \mathrm{Cic}_n $ e' bipartito). L'autovalore massimo e' $ \lambda_0 = 2 $ (il grado), con autovettore costante $ \chi_0 \equiv 1 $.
}


\section{Lemma delle strette di mano e cammini euleriani}

\subsection{Grado e lemma delle strette di mano}

\dfn{Grado di un vertice}{\label{def:grado}
  Sia $ G = (V, E, \varphi) $ un grafo e $ v \in V $. Il \textit{grado} di $ v $ e'
  \[
    \deg(v) := \#\{\text{estremi dei lati di } E \text{ incidenti a } v\}\,,
  \]
  contando ogni loop in $ v $ \textbf{due volte} (un loop in $ v $ ha entrambi gli estremi in $ v $).
}

\mlenma{Lemma delle strette di mano}{\label{lem:strette-mano}
  Per ogni grafo finito $ G = (V, E, \varphi) $:
  \[
    \sum_{v \in V} \deg(v) = 2\, |E|\,.
  \]
}
\pf{Dim}{
  Ogni lato $ e \in E $ ha esattamente due estremi (eventualmente coincidenti se $ e $ e' un loop): contribuisce dunque due volte alla somma a sinistra. Sommando su tutti i lati:
  \[
    \sum_{v \in V} \deg(v) = \sum_{e \in E} 2 = 2|E|\,.
  \]
}

\thm{Parita' dei vertici di grado dispari}{\label{thm:parita-grado-dispari}
  In ogni grafo finito, il numero di vertici di grado dispari e' pari.
}
\pf{Dim}{
  Sia $ V_1 := \{v \in V \mid \deg(v) \text{ dispari}\} $ e $ V_2 := \{v \in V \mid \deg(v) \text{ pari}\} $, $ V = V_1 \sqcup V_2 $. Dal Lemma~\ref{lem:strette-mano}:
  \[
    \underbrace{\sum_{v \in V_1} \deg(v)}_{\text{addendi dispari}} + \underbrace{\sum_{v \in V_2} \deg(v)}_{\text{pari}} = 2|E| \quad \text{(pari)}\,.
  \]
  Il secondo addendo a sinistra e' pari (somma di numeri pari) e il totale e' pari, quindi $ \sum_{v \in V_1} \deg(v) $ e' pari. Una somma di numeri dispari e' pari sse il numero di addendi e' pari: dunque $ |V_1| $ e' pari.
}

\ex{Esercizio (la festa)}{
  Ad una festa con $ 51 $ invitati, dimostrare che esiste almeno una persona che conosce un numero \textbf{pari} di altri invitati.

  \textit{Soluzione.} Costruiamo il grafo $ G $ con vertici = invitati e un lato fra $ u $ e $ v $ sse $ u $ e $ v $ si conoscono (relazione simmetrica). Allora $ \deg(v) $ = numero di persone che $ v $ conosce. Per il Teorema~\ref{thm:parita-grado-dispari} il numero di vertici di grado dispari e' pari; ma $ |V| = 51 $ e' dispari, quindi almeno un vertice ha grado pari.
}

\subsection{Cammini euleriani}

\dfn{Cammino euleriano}{\label{def:cammino-euleriano}
  Un \textit{cammino euleriano} in un grafo connesso $ G $ e' un cammino che attraversa ogni lato di $ G $ \textbf{esattamente una volta}. E' detto \textit{chiuso} se il vertice di partenza coincide con quello di arrivo.
}

\thm{Teorema di Eulero sui cammini euleriani}{\label{thm:eulero-cammini}
  Sia $ G $ un grafo connesso. Allora:
  \begin{itemize}
    \item \textbf{(a)} se $ G $ ha piu' di $ 2 $ vertici di grado dispari, allora $ G $ \textbf{non} ammette cammini euleriani;
    \item \textbf{(b)} se $ G $ ha esattamente $ 2 $ vertici di grado dispari, allora $ G $ ammette un cammino euleriano, ed ogni tale cammino inizia in uno dei due vertici di grado dispari e termina nell'altro;
    \item \textbf{(c)} se $ G $ non ha vertici di grado dispari (tutti pari), allora $ G $ ammette un cammino euleriano, ed ogni tale cammino e' chiuso.
  \end{itemize}
}

\nt{
  Per il Teorema~\ref{thm:parita-grado-dispari} il numero di vertici di grado dispari e' sempre pari, quindi i casi (a), (b), (c) coprono tutte le possibilita' (0, 2, oppure $ \geq 4 $ vertici di grado dispari).
}

\pf{Dim (parte necessaria: (a) e le seconde parti di (b), (c))}{
  Supponiamo che $ G $ ammetta un cammino euleriano $ \gamma $. Sia $ v $ un vertice di $ G $: ogni volta che $ \gamma $ attraversa $ v $ (senza esservi estremo) consuma esattamente $ 2 $ lati incidenti a $ v $ (uno in entrata, uno in uscita); inoltre $ \gamma $, in quanto cammino, ha due estremi.

  Distinguiamo:
  \begin{itemize}
    \item Se $ v $ \textbf{non} e' estremo di $ \gamma $: tutti i lati incidenti a $ v $ sono consumati a coppie, dunque $ \deg(v) $ e' \textbf{pari}.
    \item Se $ v $ e' estremo di $ \gamma $ (vertice iniziale o finale): il cammino consuma un numero dispari di lati incidenti a $ v $ (le coppie del transito piu' uno solitario per l'estremo), dunque $ \deg(v) $ e' \textbf{dispari} --- a meno che $ v $ sia sia estremo iniziale che finale, nel qual caso $ \gamma $ e' chiuso e $ \deg(v) $ torna pari.
  \end{itemize}

  In particolare, un vertice di grado dispari deve essere estremo di $ \gamma $ (ed estremo solitario, non doppio). Ma un cammino ha al piu' $ 2 $ estremi distinti, quindi:
  \begin{itemize}
    \item $ G $ ha al piu' $ 2 $ vertici di grado dispari $ \Rightarrow $ (a);
    \item se $ G $ ne ha esattamente $ 2 $, $ \gamma $ inizia in uno e termina nell'altro $ \Rightarrow $ seconda parte di (b);
    \item se $ G $ non ne ha, $ \gamma $ deve cominciare e finire nello stesso vertice $ \Rightarrow $ seconda parte di (c).
  \end{itemize}

  Resta da dimostrare la \textbf{sufficienza}: l'esistenza del cammino nei casi (b) e (c).
}

\pf{Dim (parte sufficiente: esistenza, caso (c) e riduzione di (b))}{
  \textbf{Caso (c): tutti i gradi pari.} Sia $ W $ un cammino \textbf{chiuso} di lunghezza \textbf{massima} fra quelli che percorrono ogni lato al piu' una volta (esiste perche' i lati sono finiti). Affermiamo che $ W $ e' euleriano.

  Per assurdo, supponiamo che esista un lato $ e $ non percorso da $ W $.

  \textit{Passo 1: esiste un lato non in $ W $ incidente a un vertice di $ W $.} Sia $ v $ un vertice di $ W $. Poiche' $ G $ e' connesso, c'e' un cammino da un estremo di $ e $ a $ v $; sia $ u $ il \textbf{primo} vertice di tale cammino che appartiene a $ W $. Il lato del cammino che arriva in $ u $ proviene da un vertice fuori da $ W $, quindi non e' un lato di $ W $: e' un lato non percorso, incidente a $ u \in W $.

\section{Algebra del gruppo e determinante di Dedekind}

\subsection{Identificazione $ V = \mathbb{C}[G] $}

Per legare gli operatori di convoluzione alla struttura algebrica del gruppo, identifichiamo $ V $ con l'\textit{algebra del gruppo} $ \mathbb{C}[G] $.

\dfn{Algebra del gruppo}{\label{def:algebra-gruppo}
  $ \mathbb{C}[G] $ e' lo spazio vettoriale complesso con base $ \{a : a \in G\} $ (indicizzata dagli elementi di $ G $), munito del prodotto bilineare definito sui basi da $ a \cdot b := a + b $ ed esteso per bilinearita'. Identifichiamo $ V $ con $ \mathbb{C}[G] $ via
  \[
    a \;\longleftrightarrow\; \delta_a\,,
  \]
  ossia $ g = \sum_{a \in G} g(a)\, a \in \mathbb{C}[G] $ corrisponde a $ g = \sum_{a \in G} g(a)\, \delta_a \in V $.
}

\mprop{Azione di $ T_b $ sulla base $ \{\delta_a\} $}{\label{prop:Tb-deltaa}
  Per ogni $ a, b \in G $:
  \[
    T_b(\delta_a) = \delta_{a - b}\,, \qquad T_{-b}(\delta_a) = \delta_{a + b}\,.
  \]
}
\pf{Dim}{
  $ T_b(\delta_a)(x) = \delta_a(x + b) $ vale $ 1 $ sse $ x + b = a $, cioe' $ x = a - b $; vale $ 0 $ altrimenti. Dunque $ T_b(\delta_a) = \delta_{a - b} $. Sostituendo $ b \to -b $ si ottiene la seconda formula.
}

\mprop{Moltiplicazione a sinistra in $ \mathbb{C}[G] $ = convoluzione}{\label{prop:left-mult-conv}
  Sia $ v = \sum_{b \in G} c_b \cdot b \in \mathbb{C}[G] $. L'operatore $ V \to V $ di \textbf{moltiplicazione a sinistra} per $ v $ in $ \mathbb{C}[G] $ coincide con l'operatore di convoluzione
  \[
    T_v := \sum_{b \in G} c_b\, T_{-b}\,.
  \]
}
\pf{Dim}{
  Per linearita', basta verificare l'identita' sui basi. Moltiplicando a sinistra per $ v $:
  \[
    v \cdot \delta_a = \Bigl(\sum_b c_b\, b\Bigr) \cdot a = \sum_b c_b\, (b + a) = \sum_b c_b\, \delta_{a + b}\,.
  \]
  D'altra parte, per la Proposizione~\ref{prop:Tb-deltaa}:
  \[
    T_v(\delta_a) = \sum_b c_b\, T_{-b}(\delta_a) = \sum_b c_b\, \delta_{a + b}\,.
  \]
}

\nt{
  Per il Teorema~\ref{thm:diag-gen} (con $ c_a \to c_{-a} $ per via dell'indice $ -b $) l'autovalore di $ T_v $ sul carattere $ \chi $ e':
  \[
    \lambda_\chi(v) = \sum_{b \in G} c_b\, \chi(-b)\,.
  \]
}

\subsection{Determinante di Dedekind}

L'osservazione chiave: la matrice di $ T_v $ rispetto alla base $ \{\delta_a\}_{a \in G} $ ha entrate $ c_{g - h} $ in posizione $ (g, h) $. Variando i $ c_b $, otteniamo un \textbf{polinomio} nelle $ |G| $ indeterminate $ \{c_b\} $.

\dfn{Matrice di Dedekind $ M(G) $}{\label{def:matrice-dedekind}
  Sia $ G $ un gruppo abeliano finito e $ \{X_a\}_{a \in G} $ una famiglia di indeterminate indicizzate dagli elementi di $ G $. La \textit{matrice di Dedekind} $ M(G) $ ha righe e colonne indicizzate da $ G $ e per ogni $ g, h \in G $:
  \[
    M(G)_{g, h} := X_{g - h}\,.
  \]
}

\ex{$ G = \mathbb{Z}/3\mathbb{Z} = \{0, 1, 2\} $}{
  Calcolando $ X_{-1} = X_2 $ e $ X_{-2} = X_1 $:
  \[
    M(G)
    = \begin{pmatrix}
      X_{0 - 0} & X_{0 - 1} & X_{0 - 2} \\
      X_{1 - 0} & X_{1 - 1} & X_{1 - 2} \\
      X_{2 - 0} & X_{2 - 1} & X_{2 - 2}
    \end{pmatrix}
    = \begin{pmatrix}
      X_0 & X_2 & X_1 \\
      X_1 & X_0 & X_2 \\
      X_2 & X_1 & X_0
    \end{pmatrix}\,.
  \]
  La trasposta $ M(G)^T $ e' la \textbf{matrice circolante standard}:
  \[
    M(G)^T = \begin{pmatrix}
      X_0 & X_1 & X_2 \\
      X_2 & X_0 & X_1 \\
      X_1 & X_2 & X_0
    \end{pmatrix}\,,
  \]
  in cui ogni riga e' lo shift ciclico della precedente. $ \det M(G) = \det M(G)^T $: i due determinanti coincidono.
}

\thm{Teorema di Dedekind sul determinante di gruppo}{\label{thm:dedekind}
  Sia $ G $ un gruppo abeliano finito. Vale l'identita' polinomiale
  \[
    \Delta(G) := \det M(G) = \prod_{\chi \in \widehat{G}} \Bigl(\sum_{b \in G} \chi(b)\, X_b\Bigr)\,. \tag{$\star$}
  \]
  In particolare $ \Delta(G) $ si \textbf{fattorizza in fattori lineari} nelle $ X_b $ (sul campo dei caratteri).
}

\pf{Dim}{
  \textbf{Passo 1: lettura operatoriale di $ M(G) $.} Fissiamo $ c_b \in \mathbb{C} $ per ogni $ b \in G $ e poniamo $ v := \sum_b c_b \cdot b \in \mathbb{C}[G] $. Per la Proposizione~\ref{prop:left-mult-conv}, la matrice di $ T_v $ rispetto a $ \{\delta_a\} $ ha entrata $ c_{g - h} $ in posizione $ (g, h) $:
  \[
    [T_v]_{g, h} = c_{g - h} = M(G)_{g, h}\,\bigl|_{X_a \mapsto c_a}\,.
  \]
  Quindi
  \[
    \det M(G)\bigl|_{X_a \mapsto c_a} = \det T_v\,.
  \]

  \textbf{Passo 2: calcolo del determinante via autovalori.} Per il Teorema~\ref{thm:diag-gen} e la Proposizione~\ref{prop:left-mult-conv}, $ T_v $ e' diagonalizzabile con autovalori
  \[
    \lambda_\chi = \sum_{b \in G} c_b\, \chi(-b)\,, \qquad \chi \in \widehat{G}\,.
  \]
  Il determinante e' il prodotto degli autovalori:
  \[
    \det T_v = \prod_{\chi \in \widehat{G}} \Bigl(\sum_{b \in G} c_b\, \chi(-b)\Bigr)\,. \tag{$\star\star$}
  \]

  \textbf{Passo 3: $ \chi(-b) $ diventa $ \chi(b) $.} Per ogni $ \chi $ vale $ \overline{\chi(b)} = \chi(b)^{-1} = \chi(-b) $, perche' $ \chi(b) \in S^1 $ (e' radice dell'unita'). Dunque $ \overline{\chi} : b \mapsto \chi(-b) $ e' essa stessa un carattere. L'applicazione $ \chi \mapsto \overline{\chi} $ e' una \textbf{biiezione} di $ \widehat{G} $ in se' (e' un'involuzione), quindi il prodotto rimane invariato:
  \[
    \prod_{\chi \in \widehat{G}} \Bigl(\sum_b c_b\, \chi(-b)\Bigr)
      = \prod_{\chi \in \widehat{G}} \Bigl(\sum_b c_b\, \overline{\chi}(b)\Bigr)
      = \prod_{\chi \in \widehat{G}} \Bigl(\sum_b c_b\, \chi(b)\Bigr)\,.
  \]

  \textbf{Passo 4: identita' polinomiale.} I due membri di $(\star)$ sono polinomi nelle indeterminate $ \{X_b\}_{b \in G} $. Per i passi precedenti, assumono lo stesso valore per ogni scelta numerica $ X_b = c_b \in \mathbb{C} $. Due polinomi in piu' variabili su un campo infinito che coincidono come funzioni sono uguali, dunque $(\star)$ vale come identita' fra polinomi.
}

\pf{Dim alternativa (via il Teorema Fondamentale della Teoria delle Rappresentazioni)}{
  Stanley cita questo teorema soprattutto per il suo interesse storico, come precursore della moderna teoria dei caratteri. Unendo le proprieta' dei determinanti di gruppo con il Teorema Fondamentale della Teoria delle Rappresentazioni (cfr.\ \texttt{fondamenti-rappr.tex}) si ottiene una dimostrazione piu' concettuale.

  \textbf{Setup.} Consideriamo l'algebra gruppale $ \mathbb{C}[G] $ (Definizione~\ref{def:algebra-gruppo}). Per il Teorema Fondamentale, $ \mathbb{C}[G] $ e' isomorfa come algebra a una somma diretta di algebre di matrici
  \[
    \mathbb{C}[G] \;\cong\; \bigoplus_{C} M_{n_C}(\mathbb{C})\,,
  \]
  dove $ C $ varia fra le classi di coniugio di $ G $ ed $ n_C $ e' la dimensione della rappresentazione irriducibile associata. L'isomorfismo che realizza la decomposizione e' la \textbf{Trasformata di Fourier} $ \mathcal{F} $.

  \textbf{Semplificazione nel caso abeliano.} Poiche' $ G $ e' abeliano:
  \begin{itemize}
    \item ogni classe di coniugio e' un singoletto, quindi indicizziamo per elementi del gruppo, ovvero — equivalentemente, per dualita' — per caratteri irriducibili;
    \item tutte le rappresentazioni irriducibili hanno dimensione $ n_C = 1 $.
  \end{itemize}
  L'isomorfismo si semplifica drasticamente in:
  \[
    \mathcal{F} \,:\, \mathbb{C}[G] \;\stackrel{\sim}{\longrightarrow}\; \bigoplus_{\chi \in \widehat{G}} \mathbb{C}\,. \tag{$\dagger$}
  \]

  \textbf{Lettura operatoriale di $ M(G) $.} Sia $ v = \sum_{b \in G} X_b \cdot b $ un elemento generico di $ \mathbb{C}[G] $ (con le $ X_b $ trattate come indeterminate). Per la Proposizione~\ref{prop:left-mult-conv}, $ M(G) $ e' \textbf{esattamente} la matrice dell'operatore $ T_v $ di moltiplicazione a sinistra per $ v $, scritta nella base $ \{\delta_a\}_{a \in G} $.

  \textbf{Immagine via Fourier.} Calcoliamo l'immagine di $ v $ tramite $ \mathcal{F} $. Sulla componente associata al carattere $ \chi $, $ v $ agisce per moltiplicazione come lo scalare $ \sum_{b \in G} X_b\, \chi(b) $ (e' il valore di $ v $ visto come funzione su $ \widehat{G} $):
  \[
    \mathcal{F}(v) = \bigoplus_{\chi \in \widehat{G}} \Bigl(\sum_{b \in G} X_b\, \chi(b)\Bigr)\,.
  \]

  \textbf{Determinante.} L'isomorfismo $ \mathcal{F} $ e' di algebre: preserva il prodotto. In particolare il determinante dell'operatore di moltiplicazione a sinistra per $ v $ in $ \mathbb{C}[G] $ (che e' $ \det M(G) $) coincide con il determinante della moltiplicazione per $ \mathcal{F}(v) $ nello spazio trasformato $ \bigoplus_\chi \mathbb{C} $.

  Quest'ultimo spazio e' somma diretta di componenti unidimensionali: la matrice di moltiplicazione per $ \mathcal{F}(v) $ e' \textbf{diagonale}, con $ \sum_b X_b\, \chi(b) $ sull'entrata indicizzata da $ \chi $. Il determinante e' il prodotto degli elementi diagonali:
  \[
    \det M(G) = \det \mathcal{F}(v) = \prod_{\chi \in \widehat{G}} \Bigl(\sum_{b \in G} \chi(b)\, X_b\Bigr)\,. \qed
  \]
}

\nt{
  Questa lettura algebrica spiega anche, a posteriori, l'estensione di Frobenius (vedi sotto): in un gruppo \textbf{non} abeliano la decomposizione di Wedderburn $ \mathbb{C}[G] \cong \bigoplus_\rho M_{d_\rho}(\mathbb{C}) $ ha blocchi di dimensione $ d_\rho > 1 $, e il determinante della moltiplicazione su un blocco $ M_{d_\rho}(\mathbb{C}) $ non e' lineare ma e' un polinomio di grado $ d_\rho $ elevato alla $ d_\rho $-esima potenza. Nel caso abeliano $ d_\rho = 1 $ per ogni $ \rho $ e si ritrovano fattori lineari.
}

\nt{
  \textbf{Estensione di Frobenius.} Per $ G $ \textbf{non} abeliano, Frobenius generalizzo' il risultato: con la matrice $ (X_{g h^{-1}})_{g, h \in G} $ vale
  \[
    \Delta(G) = \det(X_{g h^{-1}}) = \prod_{\rho \in \widehat{G}_{\text{irr}}} \det\!\Bigl(\sum_{g \in G} X_g\, \rho(g)\Bigr)^{d_\rho}\,,
  \]
  dove $ \widehat{G}_{\text{irr}} $ e' l'insieme delle rappresentazioni irriducibili di $ G $ e $ d_\rho $ e' il grado di $ \rho $. Per $ G $ abeliano tutte le rappresentazioni irriducibili sono $ 1 $-dim e $ d_\rho = 1 $: si ritrova $(\star)$.
}

\mprop{(Frobenius)}{\label{prop:frobenius-isom}
  Due gruppi finiti con lo stesso determinante di gruppo $ \Delta $ sono isomorfi.
}

\subsection{Il determinante circolante}

\ex{$ G = \mathbb{Z}/n\mathbb{Z} $: formula esplicita}{\label{ex:circolante}
  Sia $ \omega := e^{2\pi i / n} $ radice $ n $-esima primitiva dell'unita'. I caratteri di $ G = \mathbb{Z}/n\mathbb{Z} $ sono i morfismi $ \chi_k : G \to \mathbb{C}^\times $ individuati da $ \chi_k(1) = \omega^k $, ovvero
  \[
    \chi_k(j) = \omega^{j k}\,, \qquad k \in \{0, 1, \ldots, n - 1\}\,;
  \]
  sono $ n = |G| $ in tutto (conferma $ |\widehat{G}| = |G| $). Applicando il Teorema~\ref{thm:dedekind}:
  \[
    \Delta(\mathbb{Z}/n\mathbb{Z})
      = \prod_{k = 0}^{n - 1} \sum_{j = 0}^{n - 1} \chi_k(j)\, X_j
      = \prod_{k = 0}^{n - 1} \Bigl(\sum_{j = 0}^{n - 1} \omega^{j k}\, X_j\Bigr)\,.
  \]
  E' la celebre formula per il \textbf{determinante della matrice circolante}.
}


\section{Alberi e teorema di Cayley}

\subsection{Definizione e caratterizzazioni}

\dfn{Albero}{\label{def:albero}
  Un \textit{albero} e' un grafo $ G $ che e':
  \begin{enumerate}
    \item \textbf{connesso};
    \item \textbf{privo di cicli} (in particolare privo di loop).
  \end{enumerate}
}

\mprop{Caratterizzazioni equivalenti di albero}{\label{prop:caratterizzazioni-albero}
  Per un grafo $ G $ sono equivalenti:
  \begin{enumerate}
    \item $ G $ e' un albero;
    \item $ G $ e' connesso e cancellando un qualsiasi lato lo si disconnette;
    \item $ G $ e' privo di cicli e aggiungendo un qualsiasi lato (fra due vertici non adiacenti) si crea un ciclo.
  \end{enumerate}
}
\pf{Dim}{
  \textbf{$ (1) \Rightarrow (2) $.} $ G $ e' connesso per ipotesi. Sia $ e $ un lato di estremi $ u, v $ e supponiamo per assurdo che $ G - e $ resti connesso: allora esiste un cammino da $ u $ a $ v $ che non usa $ e $, e aggiungendo $ e $ si ottiene un ciclo --- contro l'ipotesi che $ G $ sia aciclico.

  \textbf{$ (2) \Rightarrow (1) $.} $ G $ e' connesso per ipotesi; resta da vedere che e' aciclico. Se $ G $ contenesse un ciclo, cancellando un lato $ e $ \textbf{appartenente al ciclo} il grafo resterebbe connesso (i due estremi di $ e $ restano collegati dal resto del ciclo), contro l'ipotesi $ (2) $.

  \textbf{$ (1) \Leftrightarrow (3) $.} Esercizio (analogo: l'aciclicita' e' $ (1) $; la connessione equivale a poter sempre chiudere un ciclo aggiungendo un lato).
}

\subsection{Foglie}

\dfn{Foglia}{\label{def:foglia}
  Un vertice di grado $ 1 $ si dice \textit{foglia}.
}

\nt{
  In questa sezione i cammini sono \textbf{semplici} (non ripetono lati): $ v_0\, e_1\, v_1\, e_2\, v_2 \cdots e_k\, v_k $ con $ e_{i+1} \neq e_i $.
}

\mprop{Esistenza di foglie}{\label{prop:esistenza-foglia}
  Un albero con almeno due vertici ha almeno una foglia. (Anzi, almeno due.)
}
\pf{Dim}{
  Consideriamo un cammino semplice $ v_0\, e_1\, v_1 \cdots e_k\, v_k $ che parte da un vertice qualsiasi ed e' \textbf{massimale} (non prolungabile). L'estremo $ v_k $ ha grado $ 1 $: se avesse grado $ \geq 2 $ ci sarebbe un lato incidente a $ v_k $ diverso da $ e_k $, e poiche' $ G $ e' aciclico tale lato porterebbe a un vertice \textbf{nuovo} $ v_{k+1} \neq v_0, \ldots, v_{k-1} $ (altrimenti si formerebbe un ciclo), permettendo di prolungare --- assurdo. Dunque $ v_k $ e' una foglia.

  Inoltre $ v_k \neq v_0 $ (altrimenti $ G $ avrebbe un ciclo): partendo da $ v_k $ e ripetendo il ragionamento dall'altro capo si trova una seconda foglia.
}

\subsection{Procedura di costruzione e conteggio dei lati}

\dfn{Procedura di crescita}{\label{def:procedura-crescita}
  Si costruisce un grafo cosi':
  \begin{itemize}
    \item si parte da un singolo nodo;
    \item ad ogni passo si disegna un nodo \textbf{nuovo} e lo si collega con un lato a un nodo gia' esistente.
  \end{itemize}
}

\mprop{La procedura genera tutti e soli gli alberi}{\label{prop:procedura-alberi}
  Con la procedura di crescita (Definizione~\ref{def:procedura-crescita}) si ottiene un albero, e ogni albero e' ottenibile in questo modo.
}
\pf{Dim}{
  \textbf{($ \Rightarrow $) Ogni grafo costruito e' un albero.} Per induzione su $ n = \#\{\text{vertici}\} $. Caso $ n = 1 $: ovvio. Sia $ G $ ottenuto da $ G_{n-1} $ aggiungendo un vertice $ v $ collegato a $ u \in G_{n-1} $.
  \begin{itemize}
    \item \textbf{Connesso:} due vertici $ w, z $ di $ G $ o appartengono entrambi a $ G_{n-1} $ (connesso per ipotesi induttiva), oppure uno e' $ v $, e allora ci si collega tramite $ u $.
    \item \textbf{Aciclico:} un eventuale ciclo non puo' passare per $ v $ (grado $ 1 $ per costruzione), ne' essere contenuto in $ G_{n-1} $ (aciclico per ipotesi induttiva).
  \end{itemize}

  \textbf{($ \Leftarrow $) Ogni albero e' ottenibile.} Per induzione sul numero di nodi. Dato un albero $ G $ con $ n \geq 2 $ vertici, per la Proposizione~\ref{prop:esistenza-foglia} esiste una foglia $ v $; sia $ G_{n-1} $ il grafo ottenuto cancellando $ v $ e l'unico lato $ e $ ad esso incidente. $ G_{n-1} $ e' ancora un albero (su $ n-1 $ vertici), quindi per ipotesi induttiva e' ottenibile con la procedura; applicando un ultimo passo (aggiunta di $ v $ tramite $ e $) si ottiene $ G $.
}

\cor{Numero di lati di un albero}{\label{cor:lati-albero}
  Ogni albero con $ n $ vertici ha esattamente $ n - 1 $ lati.
}
\pf{Dim}{
  Per la Proposizione~\ref{prop:procedura-alberi} l'albero si costruisce partendo da $ 1 $ nodo e $ 0 $ lati, aggiungendo $ n - 1 $ volte un nodo e un lato: in totale $ n - 1 $ lati.
}

\subsection{Alberi radicati}

\dfn{Albero radicato, padre, figli}{\label{def:albero-radicato}
  Un \textit{albero radicato} $ (G, r) $ e' un albero con un vertice distinto $ r $ detto \textit{radice}. Per ogni $ v \neq r $ esiste un unico cammino semplice da $ r $ a $ v $ (perche' $ G $ e' un albero): il vertice immediatamente precedente $ v $ su tale cammino si dice \textit{padre} di $ v $; i vertici adiacenti a $ v $ diversi dal padre si dicono \textit{figli} di $ v $.
}

\mprop{Unicita' del padre}{\label{prop:unicita-padre}
  In un albero radicato, ogni $ v \neq r $ ha un \textbf{unico} padre.
}
\pf{Dim}{
  Se $ p_1 \neq p_2 $ fossero due padri di $ v $, si avrebbero due cammini distinti da $ r $ a $ v $ (uno via $ p_1 $, l'altro via $ p_2 $), contro l'unicita' del cammino in un albero.
}

\subsection{Teorema di Cayley e codice di Prüfer}

\thm{Teorema di Cayley}{\label{thm:cayley}
  Il numero di alberi \textbf{etichettati} su $ n \geq 2 $ vertici (insieme dei vertici $ \{1, 2, \ldots, n\} $) e'
  \[
    \#\{\text{alberi etichettati su } n \text{ vertici}\} = n^{\,n-2}\,.
  \]
}

\nt{
  Valori piccoli: $ n = 2 \mapsto 2^0 = 1 $; $ n = 3 \mapsto 3^1 = 3 $; $ n = 4 \mapsto 4^2 = 16 $.
}

La dimostrazione esibisce una \textbf{biiezione} fra gli alberi etichettati su $ \{1, \ldots, n\} $ e le sequenze (stringhe) di lunghezza $ n - 2 $ a valori in $ \{1, \ldots, n\} $, che sono $ n^{n-2} $. Tale biiezione e' il \textbf{codice di Prüfer}.

\dfn{Codifica di Prüfer}{\label{def:prufer-cod}
  Dato un albero etichettato $ G $ su $ \{1, \ldots, n\} $, si costruisce la stringa $ (x_1, \ldots, x_{n-2}) $ ripetendo $ n - 2 $ volte:
  \begin{enumerate}
    \item sia $ y $ la \textbf{foglia di etichetta minima} del grafo corrente;
    \item si pone $ x_i := $ (unico) vertice adiacente a $ y $ (il suo "padre");
    \item si cancella $ y $ (e il suo lato) dal grafo.
  \end{enumerate}
  Dopo $ n - 2 $ passi restano $ 2 $ vertici: ci si ferma.
}

\ex{Codifica di un albero su $ 6 $ vertici}{\label{ex:prufer}
  Consideriamo l'albero con lati $ \{1,4\}, \{2,4\}, \{3,4\}, \{4,5\}, \{5,6\} $:
  \begin{center}
    \begin{tikzpicture}[
      vertex/.style={circle, draw, inner sep=1.4pt, font=\footnotesize},
      thick, scale=1.0
    ]
      \node[vertex] (4) at (0,0) {$4$};
      \node[vertex] (1) at (-1.4, 1) {$1$};
      \node[vertex] (2) at (-1.4, 0) {$2$};
      \node[vertex] (3) at (-1.4, -1) {$3$};
      \node[vertex] (5) at (1.6, 0) {$5$};
      \node[vertex] (6) at (3.0, 0) {$6$};
      \draw (4) -- (1); \draw (4) -- (2); \draw (4) -- (3);
      \draw (4) -- (5); \draw (5) -- (6);
    \end{tikzpicture}
  \end{center}
  Eseguiamo la codifica (foglia minima $ \to $ vicino):
  \begin{center}
  \begin{tabular}{c|c|c}
    foglia minima $ y $ & vicino $ x_i $ & resto \\ \hline
    $ 1 $ & $ 4 $ & $ \{2,4\},\{3,4\},\{4,5\},\{5,6\} $ \\
    $ 2 $ & $ 4 $ & $ \{3,4\},\{4,5\},\{5,6\} $ \\
    $ 3 $ & $ 4 $ & $ \{4,5\},\{5,6\} $ \\
    $ 4 $ & $ 5 $ & $ \{5,6\} $ \\
  \end{tabular}
  \end{center}
  Restano i vertici $ 5, 6 $. Il codice di Prüfer e' $ \boxed{(4, 4, 4, 5)} $, di lunghezza $ n - 2 = 4 $.
}

\nt{
  \textbf{Frequenza nel codice.} Ogni vertice $ v $ compare nel codice \textbf{esattamente $ \deg(v) - 1 $ volte}. In particolare le \textbf{foglie} (grado $ 1 $) \textbf{non compaiono}. Nell'esempio: $ 4 $ ha grado $ 4 $ e compare $ 3 $ volte; $ 5 $ ha grado $ 2 $ e compare $ 1 $ volta.
}

\dfn{Decodifica di Prüfer}{\label{def:prufer-decod}
  Data una stringa $ (x_1, \ldots, x_{n-2}) $ con $ x_i \in \{1, \ldots, n\} $, si ricostruisce l'albero cosi'. Sia $ U_1 := \{1, \ldots, n\} $. Per $ i = 1, \ldots, n-2 $:
  \begin{enumerate}
    \item sia $ y_i $ la \textbf{piu' piccola etichetta in $ U_i $ che non compare} in $ (x_i, x_{i+1}, \ldots, x_{n-2}) $;
    \item si aggiunge il lato $ \{y_i, x_i\} $;
    \item si pone $ U_{i+1} := U_i \setminus \{y_i\} $.
  \end{enumerate}
  Al termine $ U_{n-1} $ contiene esattamente $ 2 $ etichette: si aggiunge il lato fra di esse.
}

\ex{Decodifica di $ (4,4,4,5) $}{
  $ U_1 = \{1,2,3,4,5,6\} $.
  \begin{center}
  \begin{tabular}{c|c|c|c}
    $ i $ & coda $ (x_i, \ldots, x_4) $ & $ y_i $ (min non in coda) & lato \\ \hline
    $ 1 $ & $ (4,4,4,5) $ & $ 1 $ & $ \{1,4\} $ \\
    $ 2 $ & $ (4,4,5) $   & $ 2 $ & $ \{2,4\} $ \\
    $ 3 $ & $ (4,5) $     & $ 3 $ & $ \{3,4\} $ \\
    $ 4 $ & $ (5) $       & $ 4 $ & $ \{4,5\} $ \\
  \end{tabular}
  \end{center}
  Resta $ U_5 = \{5, 6\} $: si aggiunge $ \{5,6\} $. Si riottiene l'albero di partenza dell'Esempio~\ref{ex:prufer}.
}

\pf{Dim del Teorema di Cayley}{
  Mostriamo che la codifica (Definizione~\ref{def:prufer-cod}) e' una \textbf{biiezione} fra alberi etichettati su $ \{1, \ldots, n\} $ e stringhe $ \{1, \ldots, n\}^{n-2} $; poiche' queste ultime sono $ n^{n-2} $, segue la tesi.

  \textbf{Buona definizione e cardinalita' a ogni passo.} Nella decodifica, ad ogni passo $ i $ l'insieme $ U_i $ ha cardinalita' $ n - i + 1 $; la coda $ (x_i, \ldots, x_{n-2}) $ contiene al piu' $ n - 1 - i $ etichette distinte, dunque esiste sempre almeno una etichetta di $ U_i $ che non vi compare (e quindi un $ y_i $ ben definito). Ad ogni sequenza la decodifica associa un grafo con $ n $ vertici e $ n - 1 $ lati.

  \textbf{Il grafo decodificato e' un albero.} Ha $ n - 1 $ lati; basta mostrare che e' \textbf{aciclico} (un grafo aciclico con $ n $ vertici e $ n-1 $ lati e' un albero, cfr.\ Corollario~\ref{cor:lati-albero}). Ad ogni passo $ y_i $ viene rimosso da $ U $ e non comparira' piu' come estremo "nuovo": il lato $ \{y_i, x_i\} $ collega $ y_i $ (che non riapparira') al resto, quindi non si chiude mai un ciclo.

  \textbf{Inversa l'una dell'altra.} Codifica e decodifica sono procedure inverse: ad ogni passo la foglia minima rimossa dalla codifica coincide con il $ y_i $ scelto dalla decodifica (la piu' piccola etichetta non ancora usata e non presente nella coda residua del codice). Dunque la codifica e' iniettiva e suriettiva.
}

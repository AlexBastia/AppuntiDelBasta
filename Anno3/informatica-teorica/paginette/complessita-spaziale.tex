\chapter{Complessita' Spaziale}

Una macchina limitata in \textbf{spazio} polinomiale e' strettamente piu' potente di una limitata in \textbf{tempo} polinomiale: in tempo $t(n)$ si toccano al piu' $t(n)$ celle, quindi il tempo vincola anche lo spazio; viceversa, una macchina con spazio polinomiale puo' usare tempo esponenziale riutilizzando le stesse celle.

\section{Modello: MdT con nastro di input read-only}

Per definire in modo sensato lo spazio logaritmico, usiamo una MdT con due nastri:
\begin{itemize}
  \item \textbf{Nastro di input} (sola lettura): la testina puo' muoversi ma non scrivere.
  \item \textbf{Nastro di lavoro} (lettura/scrittura): qui la MdT svolge i calcoli.
\end{itemize}

Il vincolo di spazio si applica \textbf{solo al nastro di lavoro}. Senza questa separazione, anche uno spazio logaritmico occuperebbe almeno $n$ celle (l'input stesso), rendendo impossibile definire classi sub-lineari.

\dfn{Computation space}{
  Il \textit{computation space} di $M$ su input $w$ e' il numero di celle distinte visitate sul nastro di lavoro. Per MdTN, e' il massimo su tutti i rami di computazione.
}

\dfn{Space function}{
  Una funzione $s : \mathbb{N} \to \mathbb{N}$ e' una \textit{space function} se e' strettamente positiva e non decrescente.
}

\dfn{Running space}{
  Sia $s(n)$ una space function. Il \textit{running space} di $M$ e' $s(n)$ se per ogni $w$, a parte un numero finito di eccezioni, il computation space di $M$ su $w$ e' limitato da $s(|w|)$.

  Cioe': $M$ ``usa spazio $s(n)$'' se su \textbf{quasi tutti} gli input di lunghezza $n$ occupa al piu' $s(n)$ celle sul worktape.
}

\dfn{DSpace}{
  Sia $s(n)$ una space function.
  \[
    DSpace(s(n)) = \{ L \mid \exists\, M \text{ deterministica con } \mathcal{L}(M) = L \text{ e running space } O(s(n)) \}
  \]
}

\dfn{NSpace}{
  Sia $s(n)$ una space function.
  \[
    NSpace(s(n)) = \{ L \mid \exists\, N \text{ non deterministica con } \mathcal{L}(N) = L \text{ e running space } O(s(n)) \}
  \]
}

\section{Le classi L e NL}

\dfn{L e NL}{
  \[
    L = DSpace(\log n) \qquad NL = NSpace(\log n)
  \]
  Lo spazio logaritmico e' sufficiente per memorizzare un numero costante di \textbf{puntatori} o \textbf{contatori} sull'input: $\log n$ bit bastano per un indice in $\{1,\ldots,n\}$.
}

Chiaramente $L \subseteq NL$, ma non sappiamo se $L = NL$ (domanda aperta analoga a $P$ vs $NP$).

\ex{$L_{01} = \{0^n 1^n \mid n \geq 0\} \in L$}{
  Il linguaggio delle stringhe con $n$ zeri seguiti da $n$ uni. L'approccio ``naivo'' (copiare l'input sul worktape e fare il ping-pong marcando uno zero a sinistra e un uno a destra) funziona ma usa \textbf{spazio lineare}, perche' riscrive tutto l'input sul nastro di lavoro. Mostrare che $L_{01} \in L$ richiede un algoritmo piu' furbo.
}
\pf{Soluzione}{
  Costruiamo una MdT che usa due contatori \textbf{in binario} sul worktape.

  \begin{algorithm}[H]
    \DontPrintSemicolon
    \caption{$L_{01} \in L$}
    \KwIn{Stringa $w$}
    \lIf{$w = \varepsilon$}{\KwRet \textsc{accetta}}
    $c_0 \leftarrow 0$\;
    \While{il simbolo corrente e' $0$}{
      $c_0 \leftarrow c_0 + 1$ \tcp*{in binario sul worktape}
      avanza testina\;
    }
    $c_1 \leftarrow 0$\;
    \While{il simbolo corrente non e' blank}{
      \lIf{il simbolo corrente e' $0$}{\KwRet \textsc{rifiuta}}
      $c_1 \leftarrow c_1 + 1$\;
      avanza testina\;
    }
    \lIf{$c_0 = c_1$}{\KwRet \textsc{accetta}} \lElse{\KwRet \textsc{rifiuta}}
  \end{algorithm}

  \textbf{Analisi spaziale}. Nel caso peggiore, ciascun contatore deve rappresentare il numero $n = |w|$. Scritto in \textbf{binario}, ogni contatore occupa $\lceil \log_2 n \rceil$ celle. Poiche' i contatori sono in numero \textbf{costante} (due) e le operazioni di incremento e confronto richiedono spazio costante aggiuntivo:
  \[
    S_M(n) \in O(\log n) \quad\Longrightarrow\quad L_{01} \in L
  \]

  \textbf{Nota}: la scelta della base \emph{non e' accidentale}. Un contatore in unario occuperebbe $n$ celle (spazio lineare); in binario bastano $\log n$ bit. Un algoritmo logspace puo' ``ricordare'' indici e contatori sull'input, ma non il contenuto dell'input stesso.
}

\ex{$\textit{Reach} \in NL$}{
  \[
    \textit{Reach} = \{ \langle G, s, t \rangle \mid G \text{ grafo orientato, esiste un cammino da } s \text{ a } t \}
  \]
  L'algoritmo classico (BFS/DFS) decide $\textit{Reach}$ in tempo polinomiale, ma usa una coda/stack che nel caso pessimo contiene tutti i nodi $\Rightarrow$ spazio \textbf{lineare}. Mostrare $\textit{Reach} \in NL$ richiede un approccio non deterministico.
}
\pf{Soluzione}{
  \textbf{Idea}. Invece di memorizzare l'intero cammino, indoviniamo \textbf{un passo alla volta} e sovrascriviamo il nodo precedente. Per evitare cicli infiniti, limitiamo il numero di passi a $|V|$: se esiste un cammino, ne esiste uno di lunghezza $\leq |V|$.

  \begin{algorithm}[H]
    \DontPrintSemicolon
    \caption{$\textit{Reach} \in NL$}
    \KwIn{Grafo $G = (V, A)$, vertici $s, t \in V$}
    $p \leftarrow s$\;
    $\mathit{cnt} \leftarrow 0$\;
    \While{$\mathit{cnt} \leq |V|$}{
      \lIf{$p = t$}{\KwRet \textsc{accetta}}
      Indovina non deterministicamente $p' \in V$\;
      \lIf{$(p, p') \notin A$}{\KwRet \textsc{rifiuta}}
      $p \leftarrow p'$\;
      $\mathit{cnt} \leftarrow \mathit{cnt} + 1$\;
    }
    \KwRet \textsc{rifiuta} \tcp*{loop rilevato}
  \end{algorithm}

  \textbf{Correttezza}. Se esiste un cammino $s \to v_1 \to \cdots \to t$ di lunghezza $\leq |V|$, un ramo non deterministico lo indovina (indovinando $v_1, v_2, \ldots, t$ in sequenza) e accetta. Se non esiste, nessun ramo puo' accettare: o ci si blocca su un arco inesistente, o il contatore supera $|V|$.

  \textbf{Analisi spaziale}. Il worktape contiene solo tre valori, ciascuno di taglia logaritmica:
  \begin{itemize}
    \item il nodo corrente $p$: $\lceil \log_2 |V| \rceil$ bit (indice in $\{1, \ldots, |V|\}$),
    \item il guess $p'$: $\lceil \log_2 |V| \rceil$ bit,
    \item il contatore $\mathit{cnt}$: $\lceil \log_2 |V| \rceil$ bit.
  \end{itemize}
  Totale: $O(\log n)$. Dunque $\textit{Reach} \in NL$.
}

\subsection{NL-completezza}

Per decidere $\textit{Reach}$ il non determinismo e' davvero necessario? Esiste un algoritmo \textbf{deterministico} in logspace? \textbf{Non si sa}: sappiamo $L \subseteq NL$ ma non se $L = NL$. E' una domanda aperta, analoga in scala ridotta a $P$ vs $NP$.

Come per $NP$, sono stati individuati i ``problemi piu' difficili'' della classe: gli \textbf{NL-completi}. Vedremo che $\textit{Reach}$ stesso e' NL-completo, il che e' un forte indizio (non una prova) che $\textit{Reach} \notin L$.

\subsubsection{Perche' non usare la riduzione polinomiale?}

Per NP-completezza avevamo usato la riduzione polinomiale ($\leq_P$). Per NL non possiamo usarla: sarebbe \textbf{troppo potente}. Infatti $NL \subseteq P$ (come vedremo), quindi una riduzione in tempo polinomiale da $L \in NL$ a qualsiasi problema potrebbe ``risolvere'' $L$ direttamente durante la costruzione dell'istanza. Servirebbe una riduzione piu' debole.

\subsubsection{Trasduttore logspace}

Per le riduzioni polinomiali avevamo usato trasduttori (MdT con input read-only, worktape, output). Qui usiamo un trasduttore analogo, con un vincolo aggiuntivo sull'output:

\dfn{Trasduttore logspace}{
  Una MdT con tre nastri:
  \begin{itemize}
    \item \textbf{Nastro di input}: sola lettura, testina bidirezionale.
    \item \textbf{Nastro di lavoro}: lettura/scrittura, bidirezionale, spazio \textbf{logaritmico}.
    \item \textbf{Nastro di output}: \textbf{sola scrittura}, testina \textbf{unidirezionale} (puo' solo avanzare).
  \end{itemize}
  Il trasduttore \textit{calcola} una funzione $f$ se per ogni input $w$ lascia $f(w)$ sul nastro di output.
}

\nt{
  Il nastro di output e' write-only unidirezionale altrimenti diventerebbe un secondo foglio di brutta, e il vincolo di spazio sarebbe inutile. L'output e' ``il foglio di bella'': scritto una volta per tutte, senza ripensamenti.
}

\dfn{Riduzione logspace}{
  Siano $A, B$ due linguaggi. Esiste una \textit{riduzione logspace} da $A$ a $B$, denotata $A \leq_L B$, se esiste $f$ tale che:
  \begin{enumerate}
    \item $f$ e' calcolabile in logspace deterministico (tramite un trasduttore come sopra).
    \item Per ogni $w$: $w \in A \iff f(w) \in B$.
  \end{enumerate}
}

Le riduzioni logspace godono delle stesse proprieta' di quelle polinomiali: transitivita', chiusura sotto $L$ e $NL$, ecc.

\dfn{NL-completezza}{
  Un linguaggio $L$ e' \textit{NL-completo} se:
  \begin{enumerate}
    \item $L \in NL$ (membership).
    \item $\forall L' \in NL,\; L' \leq_L L$ (NL-hardness).
  \end{enumerate}
}

\subsection{$\textit{Reach}$ e' NL-completo}

Come SAT e' il problema ``canonico'' per NP (Cook-Levin), $\textit{Reach}$ gioca lo stesso ruolo per NL.

\thm{$\textit{Reach}$ e' NL-completo}{
  \label{th:reach-nl-completo}
  $\textit{Reach} \in NL$ e per ogni $L \in NL$, $L \leq_L \textit{Reach}$.
}
\pf{Dimostrazione}{
  \textbf{Membership}: gia' vista ($\textit{Reach} \in NL$).

  \textbf{Hardness}. Sia $L \in NL$ generico, deciso da una MdTN $M_L$ in spazio logaritmico. Costruiamo una riduzione logspace $f$ che su input $w$ produce una tripla $\langle G, s, t \rangle$ tale che:
  \[
    w \in L \iff \text{esiste un cammino da } s \text{ a } t \text{ in } G
  \]

  \textbf{Idea chiave}: i nodi di $G$ sono le \textbf{configurazioni} di $M_L$ su $w$; gli archi rappresentano i passi della funzione di transizione. Cosi' un cammino in $G$ corrisponde a una computazione accettante di $M_L$.

  \textbf{Configurazione di $M_L$ su $w$}. Poiche' l'input e' read-only, non cambia mai: non serve ricordarlo nella configurazione. Una configurazione e' data da:
  \begin{itemize}
    \item $H_1$: posizione della testina sull'input, in $\{1, \ldots, n\}$ $\to$ $O(\log n)$ bit.
    \item $H_2$: posizione della testina sul worktape, in $\{1, \ldots, O(\log n)\}$ $\to$ $O(\log \log n)$ bit.
    \item Contenuto del worktape: stringa di lunghezza $O(\log n)$ su $\Gamma$ $\to$ $O(\log n)$ bit.
    \item $Q$: stato corrente $\to$ $O(1)$ bit.
  \end{itemize}
  Scriviamo una configurazione come stringa $\langle H_1 \mid H_2 \mid \text{worktape} \mid Q \rangle$ di taglia $O(\log n)$.

  \textbf{Numero di configurazioni}: ogni configurazione ha $O(\log n)$ simboli, quindi il numero totale e' al piu' $2^{O(\log n)} = n^{O(1)}$, \textbf{polinomiale} in $n$. (E' per questo che $NL \subseteq P$: BFS deterministico su $G$ decide $L$ in tempo polinomiale.)

  \textbf{Costruzione di $G$ in logspace}. Il trasduttore \textbf{non} puo' generare tutto $G$ sul worktape: occuperebbe spazio polinomiale. Genera invece nodi e archi \textbf{uno alla volta}, scrivendoli in output:

  \textbf{Fase 1 -- nodi}. Itera su tutte le possibili stringhe di lunghezza $O(\log n)$ (come un contatore in base $|\Gamma|$); per ciascuna verifica in loco se e' una configurazione ben formata (posizioni in range, stato valido, worktape ben formato); se si', la scrive in output. Aggiunge infine un nodo fittizio $v^*$ che fara' da $t$.

  \textbf{Fase 2 -- archi di transizione}. Ripulisce il worktape. Itera su tutte le coppie $(C_1, C_2)$ di stringhe candidate, ciascuna di taglia $O(\log n)$ $\Rightarrow$ coppia di taglia $O(\log n)$. Per ogni coppia verifica: $C_1$ e $C_2$ sensate? $C_2$ e' legal successor di $C_1$ secondo $\delta$? Se si', scrive in output l'arco $(C_1, C_2)$.

  \textbf{Fase 3 -- archi verso $v^*$}. Ripulisce il worktape. Itera su tutte le configurazioni; per ogni $C$ con stato accettante scrive in output l'arco $(C, v^*)$.

  \begin{algorithm}[H]
    \DontPrintSemicolon
    \caption{Trasduttore $f$: costruzione di $\langle G, s, t\rangle$ in logspace}
    \KwIn{Stringa $w$ (read-only), descrizione di $M_L$ con $\delta$, $Q$, $\Gamma$}
    \KwOut{$\langle G, s, t\rangle$ sul nastro di output (write-only)}
    $s \leftarrow$ configurazione iniziale di $M_L$ su $w$; scrivi $s$\;
    \BlankLine
    \tcp{Fase 1 -- nodi}
    \ForEach{stringa $C$ di lunghezza $O(\log n)$ su $\Gamma$}{
      \lIf{$C$ codifica una configurazione ben formata}{scrivi $C$}
    }
    scrivi il nodo fittizio $v^*$\;
    \BlankLine
    \tcp{Fase 2 -- archi di transizione}
    ripulisci il worktape\;
    \ForEach{coppia $(C_1, C_2)$ di stringhe di lunghezza $O(\log n)$}{
      \lIf{$C_1, C_2$ ben formate \textbf{e} $C_2$ \emph{legal successor} di $C_1$ secondo $\delta$}{scrivi $(C_1, C_2)$}
    }
    \BlankLine
    \tcp{Fase 3 -- archi verso $v^*$}
    ripulisci il worktape\;
    \ForEach{configurazione $C$}{
      \lIf{$C$ ha stato accettante}{scrivi $(C, v^*)$}
    }
    $t \leftarrow v^*$\;
  \end{algorithm}

  Il worktape contiene sempre al piu' due label di configurazione piu' contatori ausiliari: spazio $O(\log n)$. L'output cresce polinomialmente ma e' write-only unidirezionale, quindi non conta.

  \textbf{Output finale}: $f(w) = \langle G, s, t \rangle$ dove $s$ e' la configurazione iniziale di $M_L$ su $w$ e $t = v^*$.

  \textbf{Correttezza}: per costruzione,
  \[
    M_L \text{ accetta } w \iff \exists \text{ cammino da } s \text{ a } v^* \text{ in } G \iff \langle G, s, t \rangle \in \textit{Reach}
  \]
  cioe' $w \in L \iff f(w) \in \textit{Reach}$. Poiche' $L \in NL$ era generico, $\textit{Reach}$ e' NL-hard. Combinando con la membership: $\textit{Reach}$ e' NL-completo.
}

\section{Teorema di Savitch}

Abbiamo visto che $\textit{Reach} \in NL$: una NTM puo' decidere la raggiungibilita' usando solo spazio logaritmico. Ma il non determinismo qui e' davvero \textbf{necessario}? Nessuno sa rispondere a $\textit{Reach}$ in spazio logaritmico \textbf{deterministico} (cioe' nessuno sa se $\textit{Reach} \in L$), e nessuno sa nemmeno dimostrare che sia impossibile. La domanda $L$ vs $NL$ e' aperta come la $P$ vs $NP$.

Pero' qualcosa di sorprendente lo sappiamo: \textbf{possiamo decidere $\textit{Reach}$ in spazio $\log^2 n$ deterministico}. Cioe' la perdita di potenza nel passare da $NL$ a determinismo costa ``solo'' un quadrato sullo spazio.

Il teorema di \textbf{Savitch (1970)} generalizza questa osservazione a \textbf{qualsiasi} classe $NSpace(s(n))$ con $s(n) \in \Omega(\log n)$: il non determinismo si puo' simulare deterministicamente al costo di al piu' \textbf{un quadrato} sullo spazio. Per il tempo, invece, simulare non determinismo deterministicamente costa al piu' un \textbf{esponenziale} (banalmente). Lo spazio si comporta in modo molto piu' ``amichevole''.

\subsection{$\textit{Reach} \in DSpace(\log^2 n)$}

Sebbene non siamo in grado di dimostrare che $\textit{Reach} \in L = DSpace(\log n)$, siamo pero' in grado di dimostrare che la perdita di determinismo costa al piu' un quadrato sullo spazio.

\thm{$\textit{Reach}$ si decide deterministicamente in spazio $O(\log^2 n)$}{
  \label{th:reach-log-squared}
  \[
    \textit{Reach} \in DSpace(\log^2 n)
  \]
}
\pf{Dimostrazione}{
  \textbf{Idea: ricerca binaria sui cammini}. Abbiamo un grafo orientato e vogliamo andare da $s$ a $t$. Il trucco per riuscire a risolverlo in spazio $\log^2 n$ e' fare una specie di \textit{ricerca binaria sul cammino}:
  \begin{center}
    \textit{``Se esiste un path da $s$ a $t$ allora esiste un nodo intermedio $u$ (a meta' strada) tale che esiste un path da $s$ a $u$ \textbf{e} un path da $u$ a $t$.''}
  \end{center}

  Questa osservazione si applica ricorsivamente: se esiste un path da $s$ a $u$ di lunghezza $k$, allora esiste un nodo intermedio $u'$ tale che da $s$ a $u'$ c'e' un path di lunghezza $\lceil k/2 \rceil$ e da $u'$ a $u$ c'e' un path di lunghezza $\lfloor k/2 \rfloor$. E cosi' via.

  Si ha un approccio ricorsivo con tre casi:
  \begin{enumerate}
    \item $k = 0$: rispondiamo ``si''' se $s = t$, ``no'' altrimenti.
    \item $k = 1$: rispondiamo ``si''' se $(s, t)$ e' un arco di $G$, ``no'' altrimenti.
    \item Altrimenti: proviamo tutti i nodi $u \in V$ per vedere se esiste $s \to u$ (di lunghezza $\lceil k/2 \rceil$) \textbf{e} $u \to t$ (di lunghezza $\lfloor k/2 \rfloor$).
  \end{enumerate}

  \begin{algorithm}[H]
    \DontPrintSemicolon
    \caption{$\textsc{Exists-path}(G = \langle V, A\rangle, s, t, k)$}
    \uIf{$k = 0$}{
      \lIf{$s = t$}{\KwRet \textsc{accetta}} \lElse{\KwRet \textsc{rifiuta}}
    }
    \uElseIf{$k = 1$}{
      \lIf{$(s, t) \in A$}{\KwRet \textsc{accetta}} \lElse{\KwRet \textsc{rifiuta}}
    }
    \Else{
      \ForEach{$u \in V$}{
        \lIf{$\textsc{Exists-path}(G, s, u, \lceil k/2 \rceil) \wedge \textsc{Exists-path}(G, u, t, \lfloor k/2 \rfloor)$}{\KwRet \textsc{accetta}}
      }
      \KwRet \textsc{rifiuta}\;
    }
  \end{algorithm}

  La chiamata iniziale per decidere $\textit{Reach}(G, s, t)$ e' $\texttt{Exists-path}(G, s, t, |V|)$, perche' se un path esiste, ne esiste anche uno semplice di lunghezza al piu' $|V|$.

  \textbf{Analisi spaziale}. Per ogni chiamata ricorsiva servono tre registri: $s, t, u$ (puntatori a nodi) e $k$ (contatore). Ciascuno occupa $O(\log n)$ bit, quindi \textbf{$O(\log n)$ per ciascun frame} dello stack ricorsivo.

  Le due chiamate ricorsive per ogni $u$ \textbf{non vengono eseguite in parallelo}: prima esploriamo la prima chiamata fino in fondo, poi la seconda. In ogni momento sullo stack c'e' un solo cammino di chiamate dalla radice alla foglia attiva. Ogni livello dimezza $k$, quindi la profondita' dello stack e' $O(\log |V|) = O(\log n)$.

  \textbf{Spazio totale}:
  \[
    \text{profondita'} \cdot \text{spazio per frame} = O(\log n) \cdot O(\log n) = O(\log^2 n)
  \]
}

\subsection{Dimostrazione del Teorema di Savitch}

\thm{Savitch (1970)}{
  \label{th:savitch}
  Sia $s(n)$ una space function con $s(n) \in \Omega(\log n)$. Allora:
  \[
    NSpace(s(n)) \subseteq DSpace(s(n)^2)
  \]
}

In particolare:
\begin{itemize}
  \item $NL = NSpace(\log n) \subseteq DSpace(\log^2 n)$ (caso $s(n) = \log n$).
  \item $NPSpace = NSpace(\text{poly}(n)) \subseteq DSpace(\text{poly}(n)^2) = PSpace$.
\end{itemize}

\pf{Dimostrazione}{
  Sfruttiamo \textbf{due elementi}: la \textbf{riduzione} verso $\textit{Reach}$ costruita nel teorema di NL-completezza (Teorema~\ref{th:reach-nl-completo}) e l'algoritmo $\textsc{Exists-path}$ (Teorema~\ref{th:reach-log-squared}).

  Sia $L \in NSpace(s(n))$. Allora esiste una MdTN $M$ con $\mathcal{L}(M) = L$ e running space $s(n)$. Vogliamo decidere $L$ deterministicamente in spazio $s(n)^2$.

  \begin{algorithm}[H]
    \DontPrintSemicolon
    \caption{Decisore deterministico di $L \in NSpace(s(n))$ in spazio $s(n)^2$}
    \KwIn{Stringa $w$, MdTN $M$ con $\mathcal{L}(M) = L$ e space $s(n)$}
    sia $G$ il \emph{computation graph} di $M$ su $w$ (generato \emph{on-the-fly})\;
    sia $C_0$ la configurazione iniziale di $M$ su $w$\;
    sia $C_{\text{acc}}$ una configurazione accettante canonica\;
    \KwRet $\textsc{Exists-path}(G, C_0, C_{\text{acc}}, |V_G|)$\;
  \end{algorithm}

  \textbf{Il computation graph}. I nodi sono le \textbf{configurazioni} di $M$ su $w$; c'e' un arco da $C$ a $C'$ se $C'$ e' legal successor di $C$ secondo $\delta$. Una configurazione e' identificata da:
  \begin{center}
    \begin{tabular}{l l l}
      \textbf{Elemento} & \textbf{Spazio} & \textbf{Motivazione} \\ \hline
      Contenuto del worktape & $O(s(n))$ & per definizione di $NSpace(s(n))$: $M$ usa al piu' $s(n)$ celle \\
      Posizione testina input & $O(\log n)$ & indice in $\{1,\ldots,n\}$ \\
      Posizione testina worktape & $O(\log s(n))$ & indice in $\{1,\ldots,s(n)\}$ \\
      Stato corrente $Q$ & $O(1)$ & $|Q|$ e' costante (insieme finito) \\
    \end{tabular}
  \end{center}

  Poiche' $s(n) \in \Omega(\log n)$, il termine $O(s(n))$ \textbf{domina} tutti gli altri. Quindi la taglia di una configurazione e' $O(s(n))$.

  \textbf{Numero di nodi e archi}. Sia $\ell = |\Gamma|$. Una configurazione e' una stringa di lunghezza $O(s(n))$ su un alfabeto di taglia $\ell$: il numero di possibili stringhe e' al piu' $\ell^{O(s(n))}$, dunque
  \[
    N = O(\ell^{s(n)})
  \]
  (la costante nell'esponente viene assorbita nel big-$O$). Ogni arco e' una coppia di configurazioni, quindi il numero di archi e' al piu' $N^2 = O(\ell^{2 s(n)})$. La taglia totale del grafo e' \textbf{esponenziale} in $s(n)$.

  \textbf{Spazio dell'algoritmo}. $\textsc{Exists-path}$ applicato a un grafo di $N$ nodi usa spazio $O(\log^2 N)$. Qui $\log N = O(s(n) \log \ell) = O(s(n))$ (perche' $\log \ell$ e' costante), quindi:
  \[
    \text{spazio} = O(\log^2 N) = O(s(n)^2)
  \]

  Dunque l'algoritmo deterministico decide $L$ in spazio $O(s(n)^2)$. Quindi $L \in DSpace(s(n)^2)$, e poiche' $L$ era arbitrario in $NSpace(s(n))$:
  \[
    NSpace(s(n)) \subseteq DSpace(s(n)^2).
  \]
}

\nt{
  Esattamente come nella riduzione di NL-completezza, il computation graph \textbf{non viene materializzato in memoria}. Quando $\texttt{Exists-path}$ chiede ``esiste l'arco $(C, C')$?'', si calcola al volo se $C'$ e' legal successor di $C$ secondo $\delta$. Quando enumera ``\texttt{for each} $u \in V$'', enumera al volo le possibili configurazioni come stringhe di taglia $O(s(n))$. Lo spazio rimane $O(s(n)^2)$ -- molto inferiore alla taglia del grafo, che e' esponenziale in $s(n)$.
}

\subsection{Conseguenza: $PSpace = NPSpace$}

Definiamo le classi $PSpace$ e $NPSpace$ come l'equivalente di $P$ e $NP$, ma sullo spazio invece che sul tempo:

\dfn{PSpace e NPSpace}{
  \[
    PSpace = \bigcup_{c \geq 1} DSpace(n^c) \qquad NPSpace = \bigcup_{c \geq 1} NSpace(n^c)
  \]
}

Banalmente $PSpace \subseteq NPSpace$. Per Savitch:
\[
  NSpace(n^c) \subseteq DSpace((n^c)^2) = DSpace(n^{2c}) \subseteq PSpace
\]
Quindi $NPSpace \subseteq PSpace$, e:

\cor{$PSpace = NPSpace$}{
  Le classi spaziali polinomiali deterministiche e non deterministiche coincidono.
}

\nt{
  \textbf{Intuizione}: per tutte le classi spaziali da polinomiale a salire, la classe deterministica eguaglia la non deterministica. Il motivo profondo:
  \begin{itemize}
    \item Il \textbf{tempo}, una volta usato, e' perso e non lo possiamo riavere.
    \item Lo \textbf{spazio} possiamo \textit{riutilizzarlo}.
  \end{itemize}
  Una macchina deterministica in $PSpace$ puo' simulare una NTM in $PSpace$ \textbf{riprovando} tutti i possibili rami uno alla volta -- cancellando il worktape fra un tentativo e l'altro. Lo spazio resta lo stesso, il tempo invece esplode (ma non c'e' un vincolo!). Per il tempo questo trucco non funziona: simulare deterministicamente una NTM polinomiale richiede tempo esponenziale, che \textbf{non si puo' riciclare} come lo spazio.
}

\subsection{La mappa delle classi}

Ora che abbiamo Savitch, possiamo dare una mappa abbastanza completa delle relazioni note fra classi.

\begin{Theorem}{$L \subseteq P$}{l-subset-p}
  Ogni linguaggio in $L$ e' decidibile in tempo polinomiale.
\end{Theorem}
\pf{Dimostrazione}{
  Sia $M$ deterministica in spazio logaritmico, che si arresti sempre. Le possibili configurazioni attraversate sono al piu' $2^{O(\log n)} = n^{O(1)}$ -- polinomiali in $n$. Poiche' $M$ si arresta sempre, non puo' passare due volte per la stessa configurazione (sarebbe un loop). Avendo un numero polinomiale di configurazioni, il tempo e' polinomiale.
}

\thm{$NL \subseteq P$}{
  Ogni linguaggio in $NL$ e' decidibile in tempo polinomiale.
}
\pf{Dimostrazione}{
  $\textit{Reach}$ e' NL-completo (Teorema~\ref{th:reach-nl-completo}): ogni problema in $NL$ si riduce in logspace (e in particolare in tempo polinomiale, per il Teorema~\ref{Theorem:l-subset-p}) a $\textit{Reach}$. $\textit{Reach}$ si risolve in tempo polinomiale tramite BFS deterministico. Componendo: $NL \subseteq P$.
}

\thm{$NP \subseteq PSpace$}{
  Ogni linguaggio in $NP$ e' decidibile in spazio polinomiale.
}
\pf{Dimostrazione}{
  Sia $M$ una NTM in tempo polinomiale: in tempo polinomiale non puo' scrivere piu' di una quantita' polinomiale di celle, quindi $M$ e' anche in $NPSpace$. Per Savitch: $NP \subseteq NPSpace = PSpace$.
}

\thm{$coNP \subseteq PSpace$}{
  Ogni linguaggio in $coNP$ e' decidibile in spazio polinomiale.
}
\pf{Dimostrazione}{
  Sia $L \in coNP$. Per definizione $\overline{L} \in NP \subseteq PSpace$. $PSpace$ e' deterministica, quindi chiusa per complemento (basta scambiare accept/reject). Dunque $L \in PSpace$.
}

\thm{$PSpace \subseteq EXP$}{
  Ogni linguaggio decidibile in spazio polinomiale e' decidibile in tempo esponenziale.
}
\pf{Dimostrazione}{
  Una macchina in $PSpace$ ha al piu' $2^{\text{poly}(n)}$ configurazioni distinte. Se si arresta sempre, il tempo e' al piu' $2^{\text{poly}(n)}$. Dunque $PSpace \subseteq EXP \subseteq NEXP$.
}

\textbf{Schema riepilogativo}:
\[
  L \subseteq NL \subseteq P \subseteq NP \cup coNP \subseteq PSpace = NPSpace \subseteq EXP \subseteq NEXP
\]

\end{center}

\nt{
  Tutte le inclusioni di sopra sono \textbf{note}, ma le sole inclusioni \textbf{strette} dimostrate lungo questa catena sono:
  \[
    L \subsetneq PSpace \subsetneq EXPSPACE, \qquad P \subsetneq EXP
  \]
  (per il teorema di gerarchia spaziale e temporale rispettivamente). Tutte le altre potrebbero teoricamente essere uguaglianze, e nessuno sa.
}

\nt{
  \textbf{Problemi P-completi}. I problemi piu' difficili dentro $P$ sono i \textit{problemi P-completi} (rispetto a riduzioni logspace). Si congettura che non siano in $L$, ma non e' dimostrato. Caratterizzazione informale: i problemi P-completi sono \textbf{inerentemente sequenziali}, cioe' non parallelizzabili in modo significativo. Al contrario, i problemi in $L$ (o piu' in generale in $NC$) sono \textbf{parallelizzabili}. La classe $NC$ dei problemi efficientemente parallelizzabili sta fra $L$ e $P$, e la sua precisa relazione con esse e' una delle frontiere aperte della teoria.
}
